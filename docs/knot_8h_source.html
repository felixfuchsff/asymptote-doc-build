324 // Represents properties that can be computed along a knotlist.
325 // Examples include distances (d), turning angles (psi), and the linear
326 // equations used to solve for the thetas.
327 template <typename T>
328 class knotprop {
329 protected:
330  knotlist& l;
331 
332  // Calculate the property for the usual case in the iteration (and for a
333  // cyclic knot, the only case), at the index given.
334  virtual T mid(Int) = 0;
335 
336  // The special cases, these default to the usual case: mid.
337  virtual T solo(Int j) // Calculates the property for a list of length 0.
338  {
339  return mid(j);
340  }
341  virtual T start(Int j) // Calculates it at the start of the list.
342  {
343  return mid(j);
344  }
345  virtual T end(Int j) // Calculate it at the end.
346  {
347  return mid(j);
348  }
349 
350  virtual cvector<T> linearCompute()
351  {
352  Int n=l.length();
353  cvector<T> v;
354  if (n==0)
355  v.push_back(solo(0));
356  else {
357  v.push_back(start(0));
358  for (Int j=1; j<n; ++j)
359  v.push_back(mid(j));
360  v.push_back(end(n));
361  }
362  return v;
363  }
364 
365  virtual cvector<T> cyclicCompute()
366  {
367  Int n=l.length();
368  cvector<T> v;
369  for (Int j=0; j<n; ++j)
370  v.push_back(mid(j));
371  return v;
372  }
373 
374  virtual cvector<T> linearBackCompute()
375  {
376  Int n=l.length();
377  cvector<T> v;
378  if (n==0)
379  v.push_back(solo(0));
380  else {
381  v.push_back(end(n));
382  for (Int j=1; j<n; ++j)
383  v.push_back(mid(n-j));
384  v.push_back(start(0));
385  }
386  return v;
387  }
388 
389  virtual cvector<T> cyclicBackCompute()
390  {
391  Int n=l.length();
392  cvector<T> v;
393  for (Int j=1; j<=n; ++j)
394  v.push_back(mid(n-j));
395  return v;
396  }
397 
398 public:
399  virtual ~knotprop() {}
400 
401  virtual cvector<T> compute() {
402  return l.cyclic() ? cyclicCompute() : linearCompute();
403  }
404 
405  // Compute the values in the opposite order. This is needed for instance if
406  // the i-th calculation needed a result computed in the i+1-th, such as in the
407  // back substitution for solving thetas.
408  virtual cvector<T> backCompute() {
409  cvector<T> v=l.cyclic() ? cyclicBackCompute() : linearBackCompute();
410 
411  // Even though they are computed in the backwards order, return them in the
412  // standard order.
413  reverse(v.begin(),v.end());
414  return v;
415  }
416 
417  knotprop(knotlist& l)
418  : l(l) {}
419 };
