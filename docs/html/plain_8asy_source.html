81 restoreThunk buildRestoreThunk()
82 {
83  // Call the save functions in reverse order, storing their restore thunks.
84  restoreThunk[] thunks={};
85  for(int i=saveFunctions.length-1; i >= 0; --i)
86  thunks.push(saveFunctions[i]());
87 
88  return new void() {
89  // Call the restore thunks in an order matching the saves.
90  for(int i=thunks.length-1; i >= 0; --i)

267 cputime cputime()
268 {
269  static processtime last;
270  real[] a=_cputime();
271  cputime cputime;
272  real clock=a[4];
273  cputime.parent.user=a[0]; // Includes system time
274  cputime.parent.system=0;
275  cputime.parent.clock=clock;
276  cputime.child.user=a[2];
277  cputime.child.system=a[3];
278  cputime.child.clock=0;
279  real user=cputime.parent.user+cputime.child.user;
280  real system=cputime.parent.system+cputime.child.system;
281  cputime.change.user=user-last.user;
282  cputime.change.system=system-last.system;
283  cputime.change.clock=clock-last.clock;
284  last.user=user;
285  last.system=system;
286  last.clock=clock;
287  return cputime;
288 }
