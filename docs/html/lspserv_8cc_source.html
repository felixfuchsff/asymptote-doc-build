167  void AsymptoteLspServer::generateMissingTrees(std::string const& inputFile)
168  {
169  using extRefMap=std::unordered_map<std::string, SymbolContext*>;
170  //using extRefMapLoc = std::pair<extRefMap*, std::string>;
171  std::queue<extRefMap::iterator> procList;
172  std::unordered_set<std::string> processing;
173 
174  processing.emplace(inputFile);
175  SymbolContext* ctx=symmapContextsPtr->at(inputFile).get();
176 
177  for (auto const& locPair : ctx->getEmptyRefs())
178  {
179  procList.emplace(locPair);
180  }
181 
182  // standard BFS algorithm
183  while (not procList.empty())
184  {
185  auto it=procList.front();
186  procList.pop();
187 
188  std::string filename(it->first);
189  processing.emplace(filename);
190 
191  auto mapIt=symmapContextsPtr->find(filename);
192  if (mapIt != symmapContextsPtr->end())
193  {
194  it->second=mapIt->second.get();
195  }
196  else
197  {
198  block* blk=ifile(mem::string(filename.c_str())).getTree();
199  auto s=symmapContextsPtr->emplace(
200  filename,
201  make_unique<SymbolContext>(posInFile(1, 1), filename));
202  auto fit=std::get<0>(s);
203 
204  if(blk == nullptr)
205  {
206  // dead end. file cannot be parsed. no new paths.
207  continue;
208  }
209 
210  blk->createSymMap(fit->second.get()); // parse symbol from there.
211  // set plain.asy to plain
212  if (plainCtx != nullptr)
213  {
214  fit->second->extRefs.extFileRefs[plainFile]=plainCtx;
215  }
216 
217  // also parse its neighbors
218  for (auto const& sit : fit->second->getEmptyRefs())
219  {
220  if (processing.find(sit->first) == processing.end())
221  {
222  procList.emplace(sit);
223  }
224  else
225  {
226  // import cycles detected!
227  logWarning("Import cycles detected!");
228  }
229  }
230  it->second=fit->second.get();
231  }
232  }
233  }

454  td_documentColor::response AsymptoteLspServer::handleDocColorRequest(td_documentColor::request const& req)
455  {
456  td_documentColor::response rsp;
457 
458  if (SymbolContext* fileSymPtr=fromRawPath(req.params.textDocument))
459  {
460  logInfo("Got Document color request.");
461  auto& colorsInfo = fileSymPtr->colorInformation;
462  for (auto const& colorPtr : colorsInfo)
463  {
464  ColorInformation cif;
465 
466  cif.color = static_cast<TextDocument::Color>(*colorPtr);
467  cif.range.start=toLsPosition(colorPtr->rangeBegin);
468 
469  auto s=colorPtr->lastArgPosition;
470  auto& line=std::get<0>(s);
471  auto& colm=std::get<1>(s);
472  size_t offset = 0;
473  size_t lineOffset = 0;
474 
475  auto& strLines = fileContentsPtr->at(getDocIdentifierRawPath(req.params.textDocument));
476  char ch=strLines[line + lineOffset - 1][colm - 1 + offset];
477 
478  while (
479  ch != ')' and ch != ';'
480  and line + lineOffset <= strLines.size()
481  )
482  {
483  ++offset;
484  if (offset > strLines[line+lineOffset-1].size())
485  {
486  ++lineOffset;
487  offset = 0;
488  }
489  if (line+lineOffset <= strLines.size())
490  {
491  ch=strLines[line + lineOffset - 1][colm - 1 + offset];
492  }
493  }
494 
495  if (ch != ')' or line + lineOffset > strLines.size())
496  {
497  continue;
498  }
499 
500  cif.range.end=toLsPosition(make_pair(line+lineOffset, colm+offset+1));
501  rsp.result.emplace_back(cif);
502  }
503  }
504 
505  return rsp;
