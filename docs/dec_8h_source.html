671 // Abstract base class for
672 // from _ access _; (fromaccessdec)
673 // and
674 // from _ unravel _; (unraveldec)
675 class fromdec : public dec {
676 protected:
677  struct qualifier {
678  // The varEntry holds the location and the type of the highest framed
679  // structure that can be put on the stack. The record holds the actual
680  // type of the qualifier.
681  // For example:
682  // struct A {
683  // struct B {
684  // static int x;
685  // }
686  // }
687  // A a=new A;
688  // from a.B unravel x;
689  //
690  // Here, v->getType() will yield A and v->getLocation() will yield the
691  // location of the variable a, but the record type t will be B.
692  record *t;
693  varEntry *v;
694 
695  qualifier(record *t, varEntry *v)
696  : t(t), v(v) {}
697  };
698 
699  // Return the qualifier from which the fields are taken. If t==0, it is
700  // assumed that an error occurred and was reported.
701  virtual qualifier getQualifier(coenv &e, record *r) = 0;
702  idpairlist *fields;
703 
704 public:
705  fromdec(position pos, idpairlist *fields)
706  : dec(pos), fields(fields) {}
707 
708  void prettyprint(ostream &out, Int indent) override;
709 
710  void transAsField(coenv &e, record *r) override;
711 };
