149 // Add a function with one or more default arguments.
150 varEntry *addFunc(venv &ve, bltin f, ty *result, symbol name,
151  formal f1, formal f2, formal f3, formal f4, formal f5, formal f6,
152  formal f7, formal f8, formal f9, formal fA, formal fB, formal fC,
153  formal fD, formal fE, formal fF, formal fG, formal fH, formal fI)
154 {
155 #ifdef DEBUG_BLTIN
156  // If the function is an operator, print out the whole signature with the
157  // types, as operators are heavily overloaded. min and max are also heavily
158  // overloaded, so we check for them too. Many builtin functions have so
159  // many arguments that it is noise to print out their full signatures.
160  string s = name;
161  if (s.find("operator ", 0) == 0 || s == "min" || s == "max")
162  {
163  function *fun = functionFromFormals(result,f1,f2,f3,f4,f5,f6,f7,f8,f9,
164  fA,fB,fC,fD,fE,fF,fG,fH,fI);
165  ostringstream out;
166  fun->printVar(out, name);
167  REGISTER_BLTIN(f, out.str());
168  }
169  else {
170  REGISTER_BLTIN(f, name);
171  }
172 #endif
173 
174  access *a = new bltinAccess(f);
175  return addFunc(ve,a,result,name,f1,f2,f3,f4,f5,f6,f7,f8,f9,
176  fA,fB,fC,fD,fE,fF,fG,fH,fI);

599 // Adds standard functions for a newly added array type.
600 void addArrayOps(venv &ve, types::array *t)
601 {
602  ty *ct = t->celltype;
603 
604  // Check for the alias function to see if these operation have already been
605  // added, if they have, don't add them again.
606  static types::function aliasType(primBoolean(), primVoid(), primVoid());
607  aliasType.sig.formals[0].t = t;
608  aliasType.sig.formals[1].t = t;
609 
610  if (ve.lookByType(SYM(alias), &aliasType))
611  return;
612 
613  addFunc(ve, run::arrayAlias,
614  primBoolean(), SYM(alias), formal(t, SYM(a)), formal(t, SYM(b)));
615 
616  size_t depth=(size_t) t->depth();
617 
618  // Define an array constructor. This needs to know the depth of the array,
619  // which may not be known at runtime. Therefore, the depth, which is known
620  // here at compile-time, is pushed on the stack beforehand by use of a
621  // thunk.
622  callable *copyValueFunc = new thunk(new vm::bfunc(run::copyArrayValue),(Int) depth-1);
623  addFunc(ve, new callableAccess(copyValueFunc),
624  t, SYM(array), formal(primInt(), SYM(n)),
625  formal(ct, SYM(value)),
626  formal(primInt(), SYM(depth), true));
627 
628  callable *copyFunc = new thunk(new vm::bfunc(run::copyArray),(Int) depth);
629  addFunc(ve, new callableAccess(copyFunc),
630  t, SYM(copy), formal(t, SYM(a)), formal(primInt(), SYM(depth), true));
631 
632  addFunc(ve, run::arrayFunction,
633  t, SYM(map), formal(new function(ct, ct), SYM(f)), formal(t, SYM(a)));
634 
635  addFunc(ve, run::arraySequence,
636  t, SYM(sequence), formal(new function(ct, primInt()), SYM(f)),
637  formal(primInt(), SYM(n)));
638 
639  addFunc(ve, run::arraySort,
640  t, SYM(sort), formal(t, SYM(a)),
641  formal(new function(primBoolean(), ct, ct), SYM(less)),
642  formal(primBoolean(), SYM(stable), true));
643 
644  switch (depth) {
645  case 1:
646  addRestFunc(ve, run::arrayConcat, t, SYM(concat), new types::array(t));
647  addFunc(ve, run::arraySearch,
648  primInt(), SYM(search), formal(t, SYM(a)), formal(ct, SYM(key)),
649  formal(new function(primBoolean(), ct, ct), SYM(less)));
650  break;
651  case 2:
652  addFunc(ve, run::array2Transpose, t, SYM(transpose), formal(t, SYM(a)));
653  break;
654  case 3:
655  addFunc(ve, run::array3Transpose, t, SYM(transpose), formal(t, SYM(a)),
656  formal(IntArray(),SYM(perm)));

924 {
925  array *a=pop<array *>(Stack);
926  item itj=pop(Stack);
927  bool jdefault=isdefault(itj);
928  item iti=pop(Stack);

932  (*a).clear();
933  return;

935  } else {
936  i=get<Int>(iti);

944  return;
945  }
