49 public:
50  typedef mem::vector<T *> Pointer;
51  Pointer pointer;
52 
53  // Return first available index
54  size_t available() {
55  size_t index=0;
56  for(auto p=pointer.begin(); p != pointer.end(); ++p) {
57  if(*p == NULL) {return index;}
58  ++index;
59  }
60  pointer.push_back(NULL);
61  return index;
62  }
63 
64  size_t add(T *p) {
65  size_t index=available();
66  pointer[index]=p;
67  return index;
68  }
69 
70  void remove(size_t index) {
71  pointer[index]=NULL;
72  }
73 
74  ~terminator() {
75  for(auto p=pointer.begin(); p != pointer.end(); ++p) {
76  if(*p != NULL) {
77  (*p)->~T();
