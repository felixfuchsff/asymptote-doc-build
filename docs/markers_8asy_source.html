158  pair p0=relpoint(lpth,0), p1=relpoint(lpth,1);
159  pair ac=p0-p0-A+O, bd=p1-p1-B+O, det=(conj(ac)*bd).y;
160  pair op=(det == 0) ? O : p0+(conj(p1-p0)*bd).y*ac/det;
