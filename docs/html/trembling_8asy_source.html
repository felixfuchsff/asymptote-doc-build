86  path addnode(path g, real t)
87  {// Add a node to 'g' at point(g,t).
88  real l=length(g);
89  real rt=t % 1;
90  if(l == 0 || (t > l && !cyclic(g)) || rt == 0) return g;
91  if(cyclic(g)) t=t % l;

95  c0=postcontrol(g,t0), c1=precontrol(g,t1),
96  m0=(1-rt)*z0+rt*c0, m1=(1-rt)*c0+rt*c1,
97  m2=(1-rt)*c1+rt*z1, m3=(1-rt)*m0+rt*m1,
98  m4=(1-rt)*m1+rt*m2;
99  guide og=subpath(g,0,t0)..controls m0 and m3..point(g,t);

145  path deform(path g...pair[] magneticPoints) {
146  /* Return g as it was handwriting.
147  The postcontrols and precontrols of the nodes of g will be rotated
148  by an angle proportional to 'angle'(in degrees).
149  If frequency < 1, floor(1/frequency) nodes will be added to g to
150  increase the control points.
151  If frequency>= 1, one point for floor(frequency) will be used to deform
152  the path.
153  'random' controls the randomized coefficient which will be multiplied
154  by 'angle'.
155  random is 0 means don't use randomized coefficient;
156  The higher 'random' is, the more the trembling is randomized. */
157  if(length(g) == 0) return g;
158  g=addnodes(g,fuzz*abs(max(g)-min(g))...magneticPoints);
159  path tg=g;
160  frequency=abs(frequency);
161  int f=abs(floor(1/frequency)-1);
162  tg=addnodes(tg,f);
163  int frequency=floor(frequency);
164  int tf=(frequency == 0) ? 1 : frequency;
165  int l=length(tg);

167  random=abs(random);
