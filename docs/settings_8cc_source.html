148 /**
149  * Use key to look up an entry in the MSWindows registry,
150  * @param baseRegLocation base location for a key
151  * @param key Key to look up, respecting wild cards. Note that wildcards
152  * only support single-level glob. Recursive globs are not supported.
153  * @param value Value to look up
154  * @remark Wildcards can only be in keys, not in the final value
155  * @return Entry value, or nullopt if not found
156  */
157 optional<string>
158 getEntry(HKEY const& baseRegLocation, string const& key, string const& value)
159 {
160  string path= key;
161  if (key.find('\\') == 0) {
162  path= path.substr(1);// strip the prefix separator
163  }
164 
165  size_t const star= path.find('*');
166  if (star == string::npos) {
167  // absolute path, can return right away
168  DWORD dataSize= 0;
169  if (RegGetValueA(
170  baseRegLocation, path.c_str(), value.c_str(), RRF_RT_REG_SZ,
171  nullptr, nullptr, &dataSize
172  ) != ERROR_SUCCESS) {
173  return nullopt;
174  }
175 
176  mem::vector<BYTE> outputBuffer(dataSize);
177 
178  if (RegGetValueA(
179  baseRegLocation, path.c_str(), value.c_str(), RRF_RT_REG_SZ,
180  nullptr, outputBuffer.data(), &dataSize
181  ) != ERROR_SUCCESS) {
182  return nullopt;
183  }
184 
185  return make_optional<string>(
186  reinterpret_cast<char const*>(outputBuffer.data())
187  );
188  }
189 
190  // has a glob, search until we find one
191  string const prefix= path.substr(0, star);
192  string const pathSuffix= path.substr(star + 1);
193 
194  // open the key in prefix
195 
196  camp::w32::RegKeyWrapper directoryWithPrefix;
197  if (RegOpenKeyExA(
198  baseRegLocation, prefix.c_str(), 0, KEY_READ,
199  directoryWithPrefix.put()
200  ) != ERROR_SUCCESS) {
201  return nullopt;// prefix path does not exist, or some other error
202  }
203 
204  DWORD numSubKeys= 0;
205  DWORD longestSubkeySize= 0;
206 
207  // querying # of subkeys + their longest path length
208  if (RegQueryInfoKeyA(
209  directoryWithPrefix.getKey(), nullptr, nullptr, nullptr,
210  &numSubKeys, &longestSubkeySize, nullptr, nullptr, nullptr,
211  nullptr, nullptr, nullptr
212  ) != ERROR_SUCCESS) {
213  return nullopt;
214  }
215 
216  mem::vector<CHAR> subkeyBuffer(longestSubkeySize + 1);
217 
218  for (DWORD i= 0; i < numSubKeys; ++i) {
219  DWORD cchValue= longestSubkeySize + 1;
220 
221  // get subkey's name
222  if (RegEnumKeyExA(
223  directoryWithPrefix.getKey(), i, subkeyBuffer.data(), &cchValue,
224  nullptr, nullptr, nullptr, nullptr
225  ) != ERROR_SUCCESS) {
226  continue;
227  }
228 
229  // open the subkey
230  camp::w32::RegKeyWrapper searchKey;
231  if (RegOpenKeyExA(
232  directoryWithPrefix.getKey(), subkeyBuffer.data(), 0, KEY_READ,
233  searchKey.put()
234  ) != ERROR_SUCCESS) {
235  continue;
236  }
237 
238  // do a recursive search starting at the opened key
239  if (auto retResult= getEntry(searchKey.getKey(), pathSuffix, value);
240  retResult.has_value()) {
241  return retResult;
242  }
243  }
244  return nullopt;
245 }
