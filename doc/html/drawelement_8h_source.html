43 // Returns true if the line a--b intersects box b.
44  bool intersect(const pair& a, const pair& b) const
45  {
46  for(Int i=0; i < 4; ++i) {
47  pair A=p[i];
48  pair B=p[i < 3 ? i+1 : 0];
49  double de=(b.x-a.x)*(A.y-B.y)-(A.x-B.x)*(b.y-a.y);
50  if(de != 0.0) {
51  de=1.0/de;
52  double t=((A.x-a.x)*(A.y-B.y)-(A.x-B.x)*(A.y-a.y))*de;
53  double T=((b.x-a.x)*(A.y-a.y)-(A.x-a.x)*(b.y-a.y))*de;
54  if(0 <= t && t <= 1 && 0 <= T && T <= 1) return true;
55  }
56  }
57  return false;
58  }
59 
60  pair operator [] (Int i) const {return p[i];}
61 
62  bool intersect(const box& b) const {
63  for(Int i=0; i < 4; ++i) {
64  pair A=b[i];
65  pair B=b[i < 3 ? i+1 : 0];
66  if(intersect(A,B)) return true;
67  }
68  return false;
69  }
