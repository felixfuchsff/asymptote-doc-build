1 /*****
2  * genv.cc
3  * Andy Hammerlindl 2002/08/29
4  *
5  * This is the global environment for the translation of programs. In
6  * actuality, it is basically a module manager. When a module is
7  * requested, it looks for the corresponding filename, and if found,
8  * parses and translates the file, returning the resultant module.
9  *
10  * genv sets up the basic type bindings and function bindings for
11  * builtin functions, casts and operators, and imports plain (if set),
12  * but all other initialization is done by the local environment defined
13  * in env.h.
14  *****/
15 
16 /* Implementation of templated modules:
17  *
18  * Translating an access declaration:
19  *
20  * access Map(Key=A, Value=B) as MapAB;
21  *
22  * run encodeLevel for both A and B
23  * this should give the parent records for each struct
24  * encode pushing the *number* of parents on the stack (i.e., push a
25  * single int)
26  * encode pushing the string "Map/1234567" on the stack
27  * encode call to builtin loadTemplatedModule
28  * also save into MapAB (varinit)
29  *
30  * build list of types (or tyEntry?)
31  *
32  * also ensure names match
33  *
34  * *****
35  *
36  * At runtime, loadTemplatedModule pops the string
37  *
38  * if the module is already loaded, it pops the levels
39  * and returns the already loaded module.
40  *
41  * if the module is not loaded, it leaves the levels on the stack
42  * and calls the initializer for the templated module
43  *
44  * it might be easiest to give the number of pushed params as an argument
45  * to loadTemplatedModule (ints and strings have no push/pop)
46  *
47  * *****
48  *
49  * Translating a templated module
50  *
51  * we start translating a file with a list of (name, type) pairs
52  *
53  * for each record type,
54  * build variables for each parent level
55  * and encode bytecode to pop the parents off the stack into these vars
56  *
57  * build tyEntry for each templated type
58  * if its a record, then use the above variables as ent->v
59  *
60  * from here,
61  * translate the file as a module as usual
62  *
63  *
64  */

164 record *genv::getModule(symbol id, string filename) {

167  symbol index=symbol::literalTrans(filename);
168  record *r=imap[index];

183 record *genv::getTemplatedModule(
184  string filename,
185  mem::vector<absyntax::namedTy*>* args
186 ) {
187  checkRecursion(filename);
188 
189  types::signature* sig = new types::signature();
190  stringstream buf;
191  buf << filename << "/";
192  for (auto arg : *args) {
193  sig->add(formal(arg->t, arg->dest));
194  buf << arg->dest << "/";
195  }
196  buf << sig->handle() << "/";
197  symbol index=symbol::literalTrans(buf.str());

199  record *r=imap[index];
