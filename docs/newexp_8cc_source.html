37 types::ty *newRecordExp::transFromTyEntry(position pos, coenv &e,
38  trans::tyEntry *ent)
39 {
40  types::ty *t = ent->t;
41  if (t->kind == ty_error)
42  return t;
43  else if (t->kind != ty_record) {
44  em.error(pos);
45  em << "type '" << *t << "' is not a structure";
46  return primError();

51  em.error(pos);

57  assert(r);

90 types::ty *newArrayExp::trans(coenv &e)
91 {
92  types::ty *c = celltype->trans(e);
93  if (c->kind == ty_void) {
94  em.error(getPos());
95  em << "cannot declare array of type void";
96  return primError();
97  }
98 
99  if (dims)
100  c = dims->truetype(c);
101 
102  if (ai) {
103  ai->transToType(e, c);
104  return c;
105  } else if (dimexps || dims) {
106  if (dimexps) {
107  for (size_t i = 0; i < dimexps->size(); ++i) {
108  (*dimexps)[i]->transToType(e, types::primInt());
109  c = new types::array(c);
110  }
111  }
112  if (dims) {
113  for (size_t i = 0; i < dims->size(); ++i) {
114  e.c.encode(inst::intpush,0);
115  }
116  }
117  e.c.encode(inst::intpush,
118  (Int) ((dimexps ? dimexps->size():0)
119  + (dims ? dims->size():0)));
120  e.c.encode(inst::builtin, run::newDeepArray);
121 
122  return c;
123  } else {
124  em.compiler(getPos());
125  em << "new array expression must have either dims or dimexps";
126  return primError();
127  }
