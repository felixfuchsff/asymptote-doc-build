21 double PatternLength(double arclength, const array& pat,
22  bool cyclic, double penwidth)
23 {
24  double sum=0.0;
25 
26  size_t n=pat.size();
27  for(unsigned i=0; i < n; i ++)
28  sum += read<double>(pat,i)*penwidth;
29 
30  if(sum == 0.0) return 0.0;
31 
32  if(n % 2 == 1) sum *= 2.0; // On/off pattern repeats after 2 cycles.
33 
34  double pat0=read<double>(pat,0);
35  // Fix bounding box resolution problem. Example:
36  // asy -f pdf testlinetype; gv -scale -2 testlinetype.pdf
37  if(!cyclic && pat0 == 0) sum += 1.0e-3*penwidth;
38 
39  double terminator=(cyclic && arclength >= 0.5*sum) ? 0.0 : pat0*penwidth;
40  int ncycle=(int)((arclength-terminator)/sum+0.5);
41 
42  return (ncycle >= 1 || terminator >= 0.75*arclength) ?
43  ncycle*sum+terminator : 0.0;
44 }

52 pen adjustdash(pen& p, double arclength, bool cyclic)
53 {
54  pen q=p;
55  // Adjust dash sizes to fit arclength; also compensate for linewidth.
56  const LineType *linetype=q.linetype();
57  size_t n=linetype->pattern.size();
58 
59  if(n > 0) {
60  double penwidth=linetype->scale ? q.width() : 1.0;
61  double factor=penwidth;
62 
63  if(linetype->adjust && arclength) {
64  double denom=PatternLength(arclength,linetype->pattern,cyclic,penwidth);
65  if(denom != 0.0) factor *= arclength/denom;
66  }
67 
68  if(factor != 1.0)
69  q.adjust(max(factor,0.1));
70  }

72 }

74 // Account for square or extended pen cap contributions to bounding box.
75 void cap(bbox& b, double t, path p, pen pentype)
76 {
77  transform T=pentype.getTransform();
78 
79  double h=0.5*pentype.width();
80  pair v=p.dir(t);
81  transform S=rotate(conj(v))*shiftless(T);
82  double xx=S.getxx(), xy=S.getxy();
83  double yx=S.getyx(), yy=S.getyy();
84  double y=hypot(yx,yy);
85  if(y == 0.0) return;
86  double numer=xx*yx+xy*yy;
87  double x=numer/y;
88  pair z=shift(T)*p.point(t);
89 
90  switch(pentype.cap()) {
91  case SquareCap:
92  {
93  pair d=rotate(v)*pair(x,y)*h;
94  b += z+d;
95  b += z-d;
96  break;
97  }
98  case RoundCap:
99  {
100  b += pad(z,pentype.bounds());
101  break;
102  }
103  case ExtendedCap:
104  {
105  transform R=rotate(v);
106  double w=(xx*yy-xy*yx)/y;
107  pair dp=R*pair(x+w,y)*h;
108  pair dm=R*pair(x-w,y)*h;
109  b += z+dp;
110  b += z+dm;
111  b += z-dp;
112  b += z-dm;
113  break;
114  }
115  }

125  transform S=rotate(conj(v))*shiftless(T);
126  double xx=S.getxx(), xy=S.getxy();
127  double yx=S.getyx(), yy=S.getyy();
128  double y=hypot(yx,yy);
129  if(y == 0.0) return;
130  double numer=xx*yx+xy*yy;
131  double x=numer/y;
132  pair z=shift(T)*p.point(t);
133 
134  switch(pentype.join()) {
135  case MiterJoin:
136  {
137  double phi=angle(-p.predir(t),false)-angle(p.postdir(t),false);
138  static const double twopi=2.0*pi;
139  if(phi > pi) phi -= twopi;
140  if(phi < -pi) phi += twopi;
141  double s=sin(0.5*phi);
142  if(s != 0.0) {
143  double factor=1.0/s;
144  if(abs(factor) < pentype.miter()) {
145  b += z-rotate(v)*pair(x,y)*h*factor;
146  break;
147  }
148  }
149  // Fall through
150  }
151  case BevelJoin:
152  {
153  pair Z=pair(x,y)*h;
154  pair d=rotate(p.predir(t))*Z;
155  b += z+d;
156  b += z-d;
157  pair D=rotate(p.postdir(t))*Z;
158  b += z+D;
