153 void transDefault(coenv &e, position pos, varEntry *v, varinit *init) {
154  // This roughly translates into the statement
155  // if (isDefault(x))
156  // x=init;
157  // where x is the variable in v and isDefault is a function that tests
158  // whether x is the default argument token.
159 
160  v->encode(READ, pos, e.c);
161 
162  label end = e.c.fwdLabel();
163  e.c.useLabel(inst::jump_if_not_default, end);
164 
165  init->transToType(e, v->getType());
166  v->encode(WRITE, pos, e.c);
167  e.c.encodePop();
168 
169  e.c.defLabel(end);
170 }

251 void fundef::baseTrans(coenv &e, types::function *ft)
252 {
253  string name = id ? string(id) : string("<anonymous function>");
254 
255  // Create a new function environment.
256  coder fc = e.c.newFunction(getPos(), name, ft);
257  coenv fe(fc,e.e);
258 
259  // Translate the function.
260  fe.e.beginScope();
261  params->trans(fe);
262 
263  body->trans(fe);
264 
265  types::ty *rt = ft->result;
266  if (rt->kind != ty_void &&
267  rt->kind != ty_error &&
268  !body->returns()) {
269  em.error(body->getPos());
270  em << "function must return a value";
271  }
272 
273  fe.e.endScope();
274 
275  // Put an instance of the new function on the stack.
276  vm::lambda *l = fe.c.close();
277  e.c.encode(inst::pushclosure);
278  e.c.encode(inst::makefunc, l);
279 }
