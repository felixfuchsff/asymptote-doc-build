51 void exp::transToType(coenv &e, types::ty *target)
52 {
53  types::ty *ct=cgetType(e);
54 
55  if (equivalent(target, ct)) {
56  transAsType(e, target);
57  return;
58  }
59 
60  // See if the cast can be handled by the fastLookupCast method, which does
61  // less memory allocation.
62  if (ct->kind != ty_overloaded &&
63  ct->kind != ty_error &&
64  target->kind != ty_error) {
65  access *a = e.e.fastLookupCast(target, ct);
66  if (a) {
67  transAsType(e, ct);
68  a->encode(trans::CALL, getPos(), e.c);
69  return;
70  }
71  }
72 
73  types::ty *source = e.e.castSource(target, ct, symbol::castsym);
74  if (source==0) {
75  if (target->kind != ty_error) {
76  types::ty *sources=cgetType(e);
77  em.error(getPos());
78 
79  em << "cannot cast ";
80  if (sources->kind==ty_overloaded)
81  em << "expression";
82  else
83  em << "'" << *sources << "'";
84  em << " to '" << *target << "'";
85  }
86  }
87  else if (source->kind==ty_overloaded) {
88  if (target->kind != ty_error) {
89  em.error(getPos());
90  em << "expression is ambiguous in cast to '" << *target << "'";
91  }
92  }
93  else {
94  transAsType(e, source);
95  e.implicitCast(getPos(), target, source);
96  }
97 }

444  if (t->kind == types::ty_function)

507 types::ty *equalityExp::trans(coenv &e) {
508  // First, try to handle by normal function resolution.
509  types::ty *t = callExp::getType(e);
510  assert(t);
511  if (t->kind != ty_error)
512  return callExp::trans(e);
513 
514  // Then, check for the function equality case.
515  exp *left = (*this->args)[0].val;
516  exp *right = (*this->args)[1].val;
517 
518  types::ty *lt = left->getType(e);
519  types::ty *rt = right->getType(e);
520 
521  // TODO: decide what null == null should do.
522 
523  // Check for function == null and null == function
524  types::ty *ft = 0;
525  if (rt->kind == types::ty_null)
526  ft = uniqueFunction(lt);
527  else if (lt->kind == types::ty_null)
528  ft = uniqueFunction(rt);
529  else
530  ft = uniqueFunction(lt, rt);
531 
532 
533  if (ft) {
534  assert(ft->kind == ty_function);
535 
536  left->transToType(e, ft);
537  right->transToType(e, ft);
538  e.c.encode(inst::builtin, bltinFromName(callee->getName()));
539 
540  return primBoolean();

542  // Let callExp report a "no such function" error.
543  types::ty *t = callExp::trans(e);
544  assert(t->kind == ty_error);

920  }
921  }

924  types::ty *ft = callee->cgetType(e);
