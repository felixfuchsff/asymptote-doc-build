35 bool usableInTemplate(ty *t) {
36  assert(t);
37  if (t->primitive()) return true;
38  assert(t->kind != ty_null);
39  assert(t->kind != ty_overloaded);
40 
41  if (t->kind == ty_record) {
42  record* r= dynamic_cast<record*>(t);
43  assert(r);
44  assert(r->getLevel());
45  if (!r->getLevel()->getParent()) return false;// r is actually a module
46  if (!r->getLevel()->getParent()->getParent()) {
47  return true;// r is a top-level record, or all nestings are static
48  }
49  return false; // r is nested non-statically
50  }
51  if (t->kind == ty_function) {
52  function* f= dynamic_cast<function*>(t);
53  assert(f);
54  // Check the types of the result and all the parameters.
55  if (!usableInTemplate(f->result)) return false;
56  const signature& sig= *f->getSignature();
57  for (const types::formal& f : sig.formals) {
58  if (!usableInTemplate(f.t)) return false;
59  }
60  if (sig.hasRest() && !usableInTemplate(sig.getRest().t)) return false;
61  return true;
62  }
63  if (t->kind == ty_array) {
64  array* a= dynamic_cast<array*>(t);
65  assert(a);
66  return usableInTemplate(a->celltype);
67  }
68  // We should have already handled all the cases.
69  assert(false);
70  return false;
71 }

1096 void recordInitializer(coenv &e, symbol id, record *r, position here)
1097 {
1098  // This is almost equivalent to the code
1099  // autounravel A operator init() { return new A; }
1100  // where A is the name of the record. The "almost" is because the code below
1101  // does not add the operator init to the environment, since in practice it
1102  // would be added to the outer environment, not the record's environment.
1103  // Since it is always added *after* any code in the record, we lose nothing
1104  // by not adding it to the environment.
1105  // Additionally, the "autounravel" is made low-priority (will not override
1106  // user-defined operator init) which is possible only for built-in functions.
1107  formals formals(here);
1108  simpleName recordName(here, id);
1109  nameTy result(here, &recordName);
1110  newRecordExp exp(here, &result);
1111  returnStm stm(here, &exp);
1112  fundef fun(here, &result, &formals, &stm);
1113  assert(r);
1114  {
1115  e.c.pushModifier(AUTOUNRAVEL);
1116  function *ft = fun.transType(e, false);
1117  assert(ft);
1118 
1119  symbol initSym=symbol::opTrans("init");
1120  varinit *init=fun.makeVarInit(ft);
1121 
1122  assert(ft->kind != types::ty_inferred);
1123 
1124  varEntry *v=makeVarEntry(here, e, r, ft);
1125  r->e.addVar(initSym, v);
1126  r->e.ve.registerAutoUnravel(initSym, v, trans::AutounravelPriority::OFFER);
1127  initializeVar(here, e, v, init);

1129  }
