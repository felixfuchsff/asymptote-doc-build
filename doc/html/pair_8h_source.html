256 // Complex exponentiation
257 inline pair pow(const pair& z, const pair& w)
258 {
259  double u=w.getx();
260  double v=w.gety();
261  if(z == 0.0) return w == 0.0 ? 1.0 : 0.0;
262  double logr=0.5*log(z.abs2());
263  double th=z.angle();
264  return exp(logr*u-th*v)*expi(logr*v+th*u);
265 }
