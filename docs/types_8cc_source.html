418 // Equivalence by design does not look at the presence of default values.
419 bool equivalent(const signature *s1, const signature *s2)
420 {
421  if (s1 == s2)
422  return true;
423 
424  // Handle null signature
425  if (s1 == 0 || s2 == 0)
426  return false;
427 
428  // Two open signatures are always equivalent, as the formals are ignored.
429  if (s1->isOpen)
430  return s2->isOpen;
431  else if (s2->isOpen)
432  return false;
433 
434  if (s1->formals.size() != s2->formals.size()) {
435  return false;
436  }
437 
438  if (!std::equal(s1->formals.begin(),s1->formals.end(),s2->formals.begin(),
439  (bool (*)(const formal&,const formal&)) equivalent))
440  return false;
441 
442  if (s1->rest.t)
443  return s2->rest.t && equivalent(s1->rest, s2->rest);
444  else
445  return s2->rest.t == 0;
446 }

554 bool equivalent(const ty *t1, const ty *t2)
555 {
556  // The same pointer must point to the same type.
557  if (t1 == t2)
558  return true;
559 
560  // Ensure if an overloaded type is compared to a non-overloaded one, that the
561  // overloaded type's method is called.
562  if (t2->kind == ty_overloaded)
563  return t2->equiv(t1);
564  if (t1->kind == ty_overloaded)
565  return t1->equiv(t2);
566 
567  // Outside of overloaded types, different kinds mean different types.
568  if (t1->kind != t2->kind)
569  return false;
570 
571  return t1->equiv(t2);
