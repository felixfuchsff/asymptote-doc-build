114 private real[] makeMappingArray(int N, triple[] data, real gamma=1.) {
115  real[] x;
116  real[] y0;
117  real[] y1;
118 
119  for (int i=0; i<data.length; ++i) {
120  x.push(data[i].x);
121  y0.push(data[i].y);
122  y1.push(data[i].z);
123  };
124 
125  x = x*(N-1);
126  real[] lut = new real[N];
127  real[] xind = (N - 1) * uniform(0, 1, N-1) ** gamma;
128  int[] ind = map(new int(real xi) {return search(x, xi);}, xind);
129  ind = ind[1:N-1]; // note that the index is shifted from python
130 
131  real[] dist = (xind[1:N-1] - x[ind])/(x[ind+1] - x[ind]);
132 
133  lut[1:N-1] = dist * (y0[ind+1] - y1[ind]) + y1[ind];
134  lut[0] = y1[0];
135  lut[N-1] = y0[y0.length-1];
136  return lut;
137 }

151  pen[] palette(int NColors=256, real gamma=1.) {
152  real[] red = makeMappingArray(NColors, this.r, gamma);
153  real[] green = makeMappingArray(NColors, this.g, gamma);
154  real[] blue = makeMappingArray(NColors, this.b, gamma);

156  pen[] pal =
157  sequence(new pen(int i) {return rgb(red[i], green[i], blue[i]);},
158  NColors);
