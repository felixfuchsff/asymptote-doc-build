214 // Method of Shimazaki and Shinomoto for selecting the optimal number of bins.
215 // Shimazaki H. and Shinomoto S., A method for selecting the bin size of a
216 // time histogram, Neural Computation (2007), Vol. 19(6), 1503-1527.
217 // cf. http://www.ton.scphys.kyoto-u.ac.jp/~hideaki/res/histogram.html
218 int bins(real[] data, int max=100)

270 linefit leastsquares(real[] x, real[] y)
271 {
272  linefit L;
273  int n=x.length;
274  if(n == 1) abort("Least squares fit requires at least 2 data points");

276  real sy=sum(y);
277  real sxx=n*sum(x^2)-sx^2;
278  real sxy=n*sum(x*y)-sx*sy;
279  L.m=sxy/sxx;
280  L.b=(sy-L.m*sx)/n;
281  if(n > 2) {
282  real syy=n*sum(y^2)-sy^2;
283  if(sxx == 0 || syy == 0) return L;
284  L.r=sxy/sqrt(sxx*syy);
285  real arg=syy-sxy^2/sxx;
286  if(arg <= 0) return L;
287  real s=sqrt(arg/(n-2));

290  }
