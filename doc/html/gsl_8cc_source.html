545 void GSLrngInit(stack *s)
546 {
547  string n = pop<string>(s,string());
548  const gsl_rng_type **t;
549  if(n.empty()) t = &gsl_rng_default;
550  else {
551  for(t=GSLrngFirstType; *t!=0; ++t)
552  if(n == string((*t)->name)) break;
553  if(*t == 0) error(GSLinvalid);
554  }
555  GSLrngFree();
556  GSLrng = gsl_rng_alloc(*t);
557  if(GSLrng == 0) {
558  GSLerror=false;
559  error("insufficient memory for allocation of GSL random number generator");
560  }
561 }
