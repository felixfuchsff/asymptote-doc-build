54 // Computes the relative distance of a control point given the angles.
55 // The name is somewhat misleading as the velocity (with respect to the
56 // variable that parameterizes the path) is relative to the distance
57 // between the knots and even then, is actually three times this.
58 // The routine is based on the velocity function in Section 131 of the MetaPost
59 // code, but differs in that it automatically accounts for the tension and
60 // bounding with tension atleast.
61 double velocity(double theta, double phi, tension t)
62 {
63  static const double VELOCITY_BOUND = 4.0;
64  static const double a = sqrt(2.0);
65  static const double b = 1.0/16.0;
66  static const double c = 1.5*(sqrt(5.0)-1.0);
67  static const double d = 1.5*(3.0-sqrt(5.0));
68 
69  double st = sin(theta), ct = cos(theta),
70  sf = sin(phi), cf = cos(phi);
71 
72  double denom = t.val * (3.0 + c*ct + d*cf);
73 
74  double r = denom != 0.0 ? (2.0 + a*(st - b*sf)*(sf - b*st)*(ct-cf)) / denom
75  : VELOCITY_BOUND;
76 
77  //cerr << " velocity(" << theta << "," << phi <<")= " << r << endl;
78  if (r > VELOCITY_BOUND)
79  r = VELOCITY_BOUND;
80 
81  // Apply boundedness condition for tension atleast cases.
82  if (t.atleast)
83  {
84  double sine = sin(theta + phi);
85  if ((st >= 0.0 && sf >= 0.0 && sine > 0.0) ||
86  (st <= 0.0 && sf <= 0.0 && sine < 0.0))
87  {
88  double rmax = sf / sine;
89  if (r > rmax)
90  r = rmax;
91  }
92  }
93 
94  return r;
95 }

259  eqn mid(Int j) {
260  double lastAlpha = l[j-1].alpha();
261  double thisAlpha = l[j].alpha();
262  double thisBeta = l[j].beta();
263  double nextBeta = l[j+1].beta();
264 
265  // Values based on the linear approximation of the curvature coming
266  // into the knot with respect to theta[j-1] and theta[j].
267  double inFactor = 1.0/(thisBeta*thisBeta*d[j-1]);
268  double A = lastAlpha*inFactor;
269  double B = (3.0 - lastAlpha)*inFactor;
270 
271  // Values based on the linear approximation of the curvature going out of
272  // the knot with respect to theta[j] and theta[j+1].
273  double outFactor = 1.0/(thisAlpha*thisAlpha*d[j]);
274  double C = (3.0 - nextBeta)*outFactor;
275  double D = nextBeta*outFactor;
276 
277  return eqn(A,B+C,D,-B*psi[j]-D*psi[j+1]);
278  }

321 /* Recalculate the equations in the form:
322  * theta[j] + post * theta[j+1] = aug + w * theta[0]
323  *
324  * Used as the first step in solve cyclic equations.
325  */
326 cvector<weqn> recalc(cvector<eqn>& e)
327 {
328  Int n=(Int) e.size();
329  cvector<weqn> we;
330  weqn lasteqn(0,1,0,0,1);
331  we.push_back(lasteqn); // As a placeholder.
332  for (Int j=1; j < n; j++) {
333  // Subtract a factor of the last equation so that the first entry is
334  // zero, then proceed to scale it.
335  eqn& q=e[j];
336  lasteqn=scale(weqn(0,q.piv-q.pre*lasteqn.post,q.post,
337  q.aug-q.pre*lasteqn.aug,-q.pre*lasteqn.w));
338  we.push_back(lasteqn);
339  }
340  // To keep all of the information enocoded in the linear equations, we need
341  // to augment the computation to replace our trivial start weqn with a
342  // real one. To do this, we take one more step in the iteration and
343  // compute the weqn for j=n, since n=0 (mod n).
344  {
345  eqn& q=e[0];
346  we.front()=scale(weqn(0,q.piv-q.pre*lasteqn.post,q.post,
347  q.aug-q.pre*lasteqn.aug,-q.pre*lasteqn.w));
348  }
349  return we;
350 }

352 double solveForTheta0(cvector<weqn>& we)
353 {
354  // Solve for theta[0]=theta[n].
355  // How we do this is essentially to write out the first equation as:
356  //
357  // theta[n] = aug[0] + w[0]*theta[0] - post[0]*theta[1]
358  //
359  // and then use the next equation to substitute in for theta[1]:
360  //
361  // theta[1] = aug[1] + w[1]*theta[0] - post[1]*theta[2]
362  //
363  // and so on until we have an equation just in terms of theta[0] and
364  // theta[n] (which are the same theta).
365  //
366  // The loop invariant maintained is that after j iterations, we have
367  // theta[n]= a + b*theta[0] + c*theta[j]
368  Int n=(Int) we.size();
369  double a=0,b=0,c=1;
370  for (Int j=0;j<n;++j) {
371  weqn& q=we[j];
372  a+=c*q.aug;
373  b+=c*q.w;
374  c=-c*q.post;
375  }
376 
377  // After the iteration we have
378  //
379  // theta[n] = a + b*theta[0] + c*theta[n]
380  //
381  // where theta[n]=theta[0], so
382  return a/(1.0-(b+c));
383 }

385 cvector<double> backsubCyclic(cvector<weqn>& we, double theta0)
386 {
387  Int n=(Int) we.size();
388  cvector<double> thetas;
389  double lastTheta=theta0;
390  for (Int j=1;j<=n;++j)
391  {
392  weqn& q=we[n-j];
393  assert(q.pre == 0 && q.piv == 1);
394  double theta=-q.post*lastTheta+q.aug+q.w*theta0;
395  thetas.push_back(theta);
396  lastTheta=theta;
397  }
398  reverse(thetas.begin(),thetas.end());
399  return thetas;
400 }

461 // Once the equations have been determined, solve for the thetas.
462 cvector<double> solveThetas(knotlist& l, cvector<eqn>& e)
463 {
464  if (homogeneous(e))
465  // We are solving Ax=0, so a solution is zero for every theta.
466  return cvector<double>(e.size(),0);
467  else if (l.cyclic()) {
468  // The knotprop template is unusually unhelpful in this case, so I
469  // won't use it here. The algorithm breaks into three passes on the
470  // object. The old Asymptote code used a two-pass method, but I
471  // implemented this to stay closer to the MetaPost source code.
472  // This might be something to look at for optimization.
473  cvector<weqn> we=recalc(e);
474  INFO(we);
475  double theta0=solveForTheta0(we);
476  return backsubCyclic(we, theta0);
477  }
478  else { /* Non-cyclic case. */
479  /* First do row operations to get it into reduced echelon form. */
480  cvector<eqn> el=ref(l,e).compute();
481 
482  /* Then, do back substitution. */
483  return backsub(l,el).backCompute();
484  }

597 void encodeStraight(protopath& p, Int k, knotlist& l)
598 {
599  pair a=l.front().z;
600  double at=l.front().tout.val;
601  pair b=l.back().z;
602  double bt=l.back().tin.val;
603  pair step=(b-a)/3.0;
604 
605  if (at==1.0 && bt==1.0) {
606  p.straight(k)=true;
607  p.post(k)=a+step;
608  p.pre(k+1)=b-step;
609  p.point(k+1)=b;
610  }
611  else {
612  p.post(k)=a+step/at;
613  p.pre(k+1)=b-step/bt;
614  p.point(k+1)=b;
615  }

669 Int nextBreakpoint(knotlist& l, Int a)
670 {
671  // This is guaranteed to terminate if a is the index of a breakpoint. If the
672  // path is non-cyclic it will stop at or before the last knot which must be a
673  // breakpoint. If the path is cyclic, it will stop at or before looping back
674  // around to a which is a breakpoint.
675  Int j=a+1;
676  while (l[j].in->open())
677  ++j;
678  return j;
679 }

681 // Write out the controls for section of the form
682 // a.. control b and c ..d
683 void writeControls(protopath& p, Int a, knotlist& l)
684 {
685  // By convention, the first point will already be encoded.
686  p.straight(a)=dynamic_cast<controlSpec *>(l[a].out)->straight;
687  p.post(a)=dynamic_cast<controlSpec *>(l[a].out)->cz;
688  p.pre(a+1)=dynamic_cast<controlSpec *>(l[a+1].in)->cz;
689  p.point(a+1)=l[a+1].z;
690 }

692 // Solves a path that has all of its specifiers laid out explicitly.
693 path solveSpecified(knotlist& l)
694 {
695  protopath p(l.size(),l.cyclic());
696 
697  Int first=firstBreakpoint(l);
698  if (first==NOBREAK)
699  /* We are solving a fully cyclic path, so do it in one swoop. */
700  solveSection(p,0,l);
701  else {
702  // Encode the first point.
703  p.point(first)=l[first].z;
704 
705  // If the path is cyclic, we should stop where we started (modulo the
706  // length of the path); otherwise, just stop at the end.
707  Int last=l.cyclic() ? first+l.length()
708  : l.length();
709  Int a=first;
710  while (a!=last) {
711  if (l[a].out->controlled()) {
712  assert(l[a+1].in->controlled());
713 
714  // Controls are already picked, just write them out.
715  writeControls(p,a,l);
716  ++a;
717  }
718  else {
719  // Find the section a to b and solve it, putting the result (starting
720  // from index a into our protopath.
721  Int b=nextBreakpoint(l,a);
722  subknotlist section(l,a,b);
723  solveSection(p,a,section);
724  a=b;
725  }
726  }
727 
728  // For a non-cyclic path, the end control points need to be set.
729  p.controlEnds();
730  }
731 
732  return p.fix();
733 }

735 /* If a knot is open on one side and restricted on the other, this replaces the
736  * open side with a restriction determined by the restriction on the other
737  * side. After this, any knot will either have two open specs or two
738  * restrictions.
739  */
740 struct partnerUp : public knoteffect {
741  partnerUp(knotlist& l)
742  : knoteffect(l) {}
743 
744  void mid(Int j) {
745  knot& k=l[j];
746  if (k.in->open() && !k.out->open())
747  k.in=k.out->inPartner(k.z);
748  else if (!k.in->open() && k.out->open())
749  k.out=k.in->outPartner(k.z);
750  }
751 };
752 
753 /* Ensures a non-cyclic path has direction specifiers at the ends, adding curls
754  * if there are none.
755  */
756 void curlEnds(knotlist& l)
757 {
758  static curlSpec endSpec;
759 
760  if (!l.cyclic()) {
761  if (l.front().in->open())
762  l.front().in=&endSpec;
763  if (l.back().out->open())
764  l.back().out=&endSpec;
765  }
766 }

768 /* If a point occurs twice in a row in a knotlist, write in controls
769  * between the two knots at that point (unless it already has controls).
770  */
771 struct controlDuplicates : public knoteffect {
772  controlDuplicates(knotlist& l)
773  : knoteffect(l) {}
774 
775  void solo(Int) { /* One point ==> no duplicates */ }
776  // start is the same as mid.
777  void mid(Int j) {
778  knot &k1=l[j];
779  knot &k2=l[j+1];
780  if (!k1.out->controlled() && k1.z==k2.z) {
781  k1.out=k2.in=new controlSpec(k1.z,true);
782  }
783  }
784  void end(Int) { /* No next point to compare with. */ }
785 };

787 path solve(knotlist& l)
788 {
789  if (l.empty())
790  return path();
791  else {
792  info(cerr, "input knotlist", l);
793  curlEnds(l);
794  controlDuplicates(l).exec();
795  partnerUp(l).exec();
796  info(cerr, "specified knotlist", l);
797  return solveSpecified(l);
798  }
