89  assert(t->kind != ty_error);

110 void tryToWriteExp(coenv &e, exp *expr)
111 {
112  position pos=expr->getPos();
113  types::ty *t=expr->cgetType(e);
114 
115  if(!t) return;
116 
117  // If the original expression is bad, just print the errors.
118  // If it is a function which returns void, just call the function.
119  if (t->kind == ty_error || t->kind == ty_void) {
120  baseExpTrans(e, expr);
121  return;
122  }
123 
124  exp *callee=new nameExp(pos, symbol::trans("write"));
125  exp *call=new callExp(pos, callee, expr);
126 
127  types::ty *ct=call->getType(e);
128  if (ct->kind == ty_error || ct->kind == ty_overloaded) {
129  if (t->kind == ty_overloaded) {
130  // Translate the expr in order to print the ambiguity error first.
131  expr->trans(e);
132  em.sync(true);
133  assert(em.errors());
134 
135  // Then, write out all of the types.
136  tryToWriteTypeOfExp(t, expr);
137  }
138  else {
139  // Write the type of the expression and, since it is unique, assign it to
140  // 'operator answer' even though its value isn't printed.
141  tryToWriteTypeOfExp(t, expr);
142  storeExp(e, t, expr);
143  }
144  }
145  else if (t->kind == ty_overloaded) {
146  // If the exp is overloaded, but the act of writing makes it
147  // unambiguous, add a suffix to the output to warn the user of this.
148  exp *suffix=new nameExp(pos,
149  symbol::trans("overloadedMessage"));
150  exp *callWithSuffix=new callExp(pos,
151  callee, expr, suffix);
152 
153  if (callWithSuffix->getType(e)->kind != ty_error)
154  baseExpTrans(e, callWithSuffix);
155  else
156  baseExpTrans(e, call);
157  }
158  else {
159  // Interactive writing can proceed normally.
160  storeAndWriteExp(e, t, expr);
161  }
162 }

211 void transLoopBody(coenv &e, stm *body) {
212  // The semantics of the language are defined so that any variable declared
213  // inside a loop are new variables for each iteration of the loop. For
214  // instance, the code
215  //
216  // int f();
217  // for (int i = 0; i < 10; ++i) {
218  // int j=10*i;
219  // if (i == 5)
220  // f = new int() { return j; };
221  // }
222  // write(f());
223  //
224  // will write 50. This is implemented by allocating a new frame for each
225  // iteration. However, this can have a big performance hit, so we first
226  // translate the code without the frame, check if it needed the closure, and
227  // rewrite the code if necessary.
228 
229  label start = e.c.defNewLabel();
230 
231  // Encode a no-op, in case we need to jump over the default implementation
232  // to a special case.
233  e.c.encode(inst::nop);
234 
235  body->markTrans(e);
236 
237  // Don't re-translate if there were errors.
238  if (em.errors())
239  return;
240 
241  if (e.c.usesClosureSinceLabel(start)){
242  // Jump over the old section.
243  label end = e.c.defNewLabel();
244  e.c.encodePatch(start, end);
245 
246  // Let coder know that break and continue need to pop the frame.
247  e.c.loopPushesFrame();
248 
249  e.c.encodePushFrame();
250  body->markTrans(e);
251  e.c.encodePopFrame();
252  }
253 }

361 void extendedForStm::trans(coenv &e) {
362  // Translate into the syntax:
363  //
364  // start[] a = set;
365  // for (int i=0; i < a.length; ++i) {
366  // start var=a[i];
367  // body
368  // }
369 
370  position pos=getPos();
371 
372  // Use gensyms for the variable names so as not to pollute the namespace.
373  symbol a=symbol::gensym("a");
374  symbol i=symbol::gensym("i");
375 
376  // Get the start type. Handle type inference as a special case.
377  types::ty *t = start->trans(e, true);
378  if (t->kind == types::ty_inferred) {
379 
380  // First ensure the array expression is an unambiguous array.
381  types::ty *at = set->cgetType(e);
382  if (at->kind != ty_array) {
383  em.error(set->getPos());
384  em << "expression is not an array of inferable type";
385 
386  // On failure, don't bother trying to translate the loop.
387  return;
388  }
389 
390  // var a=set;
391  tyEntryTy tet(pos, primInferred());
392  decid dec1(pos, new decidstart(pos, a), set);
393  vardec(pos, &tet, &dec1).trans(e);
394  }
395  else {
396  // start[] a=set;
397  arrayTy at(pos, start, new dimensions(pos));
398  decid dec1(pos, new decidstart(pos, a), set);
399  vardec(pos, &at, &dec1).trans(e);
400  }
401 
402  // { start var=a[i]; body }
403  block b(pos);
404  decid dec2(pos,
405  new decidstart(pos, var),
406  new subscriptExp(pos, new nameExp(pos, a),
407  new nameExp(pos, i)));
408  b.add(new vardec(pos, start, &dec2));
409  b.add(body);
410 
411  // for (int i=0; i < a.length; ++i)
412  // <block>
413  forStm(pos,
414  new vardec(pos, new tyEntryTy(pos, primInt()),
415  new decid(pos,
416  new decidstart(pos, i),
417  new intExp(pos, 0))),
418  new binaryExp(pos,
419  new nameExp(pos, i),
420  SYM_LT,
421  new nameExp(pos,
422  new qualifiedName(pos,
423  new simpleName(pos, a),
424  symbol::trans("length")))),
425  new expStm(pos, new prefixExp(pos, new nameExp(pos, i), SYM_PLUS)),
426  new blockStm(pos, &b)).trans(e);
427 }
