33 void v3dfile::addHeaders()
34 {
35  getXDRFile() << v3dtypes::header;
36  std::vector<std::unique_ptr<AHeader>> headers;
37 
38  headers.emplace_back(make_unique<Uint32Header>(v3dheadertypes::canvasWidth, gl::fullWidth));
39  headers.emplace_back(make_unique<Uint32Header>(v3dheadertypes::canvasHeight, gl::fullHeight));
40  headers.emplace_back(make_unique<Uint32Header>(v3dheadertypes::absolute, getSetting<bool>("absolute")));
41  headers.emplace_back(make_unique<TripleHeader>(v3dheadertypes::minBound, triple(gl::Xmin, gl::Ymin, gl::Zmin)));
42  headers.emplace_back(make_unique<TripleHeader>(v3dheadertypes::maxBound, triple(gl::Xmax, gl:: Ymax, gl::Zmax)));
43  headers.emplace_back(make_unique<Uint32Header>(v3dheadertypes::orthographic, gl::orthographic));
44  headers.emplace_back(make_unique<DoubleHeader>(v3dheadertypes::angleOfView, gl::Angle));
45  headers.emplace_back(make_unique<DoubleHeader>(v3dheadertypes::initialZoom, gl::Zoom0));
46  if(gl::Shift != pair(0.0,0.0))
47  headers.emplace_back(make_unique<PairHeader>(v3dheadertypes::viewportShift, gl::Shift*gl::Zoom0));
48  headers.emplace_back(make_unique<PairHeader>(v3dheadertypes::viewportMargin, gl::Margin));
49 
50  for(size_t i=0; i < gl::nlights; ++i) {
51  size_t i4=4*i;
52  headers.emplace_back(make_unique<LightHeader>(
53  gl::Lights[i],
54  prc::RGBAColour(gl::Diffuse[i4], gl::Diffuse[i4+1], gl::Diffuse[i4+2], 1.0)
55  ));
56  }
57 
58  headers.emplace_back(make_unique<RGBAHeader>(
59  v3dheadertypes::background,
60  prc::RGBAColour(gl::Background[0],gl::Background[1],gl::Background[2],gl::Background[3])));
61 
62  headers.emplace_back(make_unique<DoubleHeader>(v3dheadertypes::zoomFactor, getSetting<double>("zoomfactor")));
63  headers.emplace_back(make_unique<DoubleHeader>(
64  v3dheadertypes::zoomPinchFactor, getSetting<double>("zoomPinchFactor")));
65  headers.emplace_back(make_unique<DoubleHeader>(
66  v3dheadertypes::zoomPinchCap, getSetting<double>("zoomPinchCap")));
67  headers.emplace_back(make_unique<DoubleHeader>(v3dheadertypes::zoomStep, getSetting<double>("zoomstep")));
68  headers.emplace_back(make_unique<DoubleHeader>(
69  v3dheadertypes::shiftHoldDistance, getSetting<double>("shiftHoldDistance")));
70  headers.emplace_back(make_unique<DoubleHeader>(
71  v3dheadertypes::shiftWaitTime, getSetting<double>("shiftWaitTime")));
72  headers.emplace_back(make_unique<DoubleHeader>(
73  v3dheadertypes::vibrateTime, getSetting<double>("vibrateTime")));
74 
75  getXDRFile() << (uint32_t) headers.size();
76  for(const auto& header : headers) {
77  getXDRFile() << header->ty << header->getWordSize(singleprecision);
78  header->writeContent(getXDRFile());
79  }
80 }

208 void v3dfile::addTriangles(size_t nP, triple const* P, size_t nN,
209  triple const* N, size_t nC,
210  prc::RGBAColour const* C, size_t nI,
211  uint32_t const (* PI)[3], uint32_t const (* NI)[3],
212  uint32_t const (* CI)[3])
213 {
214  getXDRFile() << v3dtypes::triangles;
215  getXDRFile() << (uint32_t) nI;
216 
217  getXDRFile() << (uint32_t) nP;
218  addTriples(P,nP);
219 
220  getXDRFile() << (uint32_t) nN;
221  addTriples(N,nN);
222 
223  bool explicitNI=false;
224  for(size_t i=0; i < nI; ++i) {
225  const uint32_t *PIi=PI[i];
226  const uint32_t *NIi=NI[i];
227  if(distinct(NIi,PIi)) {
228  explicitNI=true;
229  break;
230  }
231  }
232  getXDRFile() << (uint32_t) explicitNI;
233 
234  getXDRFile() << (uint32_t) nC;
235  bool explicitCI=false;
236  if(nC) {
237  addColors(C,nC);
238  for(size_t i=0; i < nI; ++i) {
239  const uint32_t *PIi=PI[i];
240  const uint32_t *CIi=CI[i];
241  if(distinct(CIi,PIi)) {
242  explicitNI=true;
243  break;
244  }
245  }
246  getXDRFile() << (uint32_t) explicitCI;
247  }
248 
249  for(size_t i=0; i < nI; ++i) {
250  const uint32_t *PIi=PI[i];
251  const uint32_t *NIi=NI[i];
252  addIndices(PIi);
253  if(explicitNI)
254  addIndices(NIi);
255  if(nC) {
256  const uint32_t *CIi=CI[i];
257  if(explicitCI)
258  addIndices(CIi);
259  }
260  }
