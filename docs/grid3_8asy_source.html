8 struct grid3 {
9  path3 axea,axeb;
10  bounds bds;
11  triple dir;
12  valuetime vt;
13  ticklocate locate;
14  void create(picture pic, path3 axea, path3 axeb, path3 axelevel,
15  real min, real max, position pos, autoscaleT t) {
16  real position=pos.position.x;
17  triple level;
18  if(pos.relative) {
19  position=reltime(axelevel,position);
20  level=point(axelevel,position)-point(axelevel,0);
21  } else {
22  triple v=unit(point(axelevel,1)-point(axelevel,0));
23  triple zerolevel=dot(-point(axelevel,0),v)*v;
24  level=zerolevel+position*v;
25  }
26  this.axea=shift(level)*axea;
27  this.axeb=shift(level)*axeb;
28  bds=autoscale(min,max,t.scale);
29  locate=ticklocate(min,max,t,bds.min,bds.max,
30  Dir(point(axeb,0)-point(axea,0)));
31  }
32 };
