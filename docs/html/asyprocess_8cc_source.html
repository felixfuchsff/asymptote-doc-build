362 // cleanLine changes a C++ style comment (//) into a C-style comment (/* */) so
363 // that comments don't absorb subsequent lines of code when multiline input is
364 // collapsed to a single line in the history.
365 //
366 // ex. if (x==1) // test x
367 // x=2;
368 // becomes
369 // if (x==1) /* test x */ x=2 (all on one line)
370 //
371 // cleanLine is a mess because we have to check that the // is not in a string
372 // or c-style comment, which entails re-inventing much of the lexer. The
373 // routine handles most cases, but multiline strings lose their newlines when
374 // recorded in the history.
375 typedef string::size_type size_type;
376 const size_type npos=string::npos;
377 
378 inline size_type min(size_type a, size_type b) {
379  return a < b ? a : b;
380 }
381 
382 // If start is an offset somewhere within a string, this returns the first
383 // offset outside of the string. sym is the character used to start the string,
384 // ' or ".
385 size_type endOfString(const char sym, const string line, size_type start) {
386  size_type endString=line.find(sym, start);
387  if (endString == npos)
388  return npos;
389 
390  size_type escapeInString=min(line.find(string("\\")+sym, start),
391  line.find("\\\\", start));
392  if (endString < escapeInString)
393  return endString+1;
394  else
395  return endOfString(sym, line, escapeInString+2);
396 }

398 // If start is an offset somewhere within a C-style comment, this returns the
399 // first offset outside of the comment.
400 size_type endOfComment(const string line, size_type start) {
401  size_type endComment=line.find("*/", start);
402  if (endComment == npos)

404  else
405  return endComment+2;

408 // Find the start of a string literal in the line.
409 size_type stringPos(const string line, size_type start) {
410  if (start == npos)
411  return npos;
412 
413  size_type pos=line.find_first_of("\'\"", start);
414  if (pos == npos)
415  return npos;
416 
417  // Skip over comments /* */ and ignore anything after //
418  size_type startComment=line.find("/*", start);
419  size_type startLineComment=line.find("//", start);
420 
421  if (min(startComment,startLineComment) < pos)
422  return stringPos(line,
423  startComment < startLineComment ?
424  endOfComment(line, startComment+2) :
425  npos);
426  else
427  // Nothing to skip over - the symbol actually starts a string.
428  return pos;
429 }

431 // A multiline string should retain its newlines when collapsed to a single
432 // line. This converts
433 // 'hello
434 // there'
435 // to
436 // 'hello\nthere'
437 // and
438 // "hello
439 // there"
440 // to
441 // "hello" '\n' "there"
442 // If the line doesn't end mid-string, this adds a space to the end to preserve
443 // whitespace, since it is the last function to touch the line.
444 string endString(const string line, size_type start) {
445  assert(start!=npos);
446 
447  size_type pos=stringPos(line, start);
448 
449  if (pos==npos)
450  // String ends in normal code.
451  return line+" ";
452  else {
453  char sym=line[pos];
454  size_type eos=endOfString(sym, line, pos+1);
455  if (eos==npos) {
456  // Line ends while in a string, attach a newline symbol.
457  switch (line[pos]) {
458  case '\'':
459  return line+"\\n";
460  case '\"':
461  return line+"\" \'\\n\' \"";
462  default:
463  assert(False);
464  return line;
465  }
466  }
467  else {
468  return endString(line, eos+1);
469  }
470  }
471 }

473 // Find the first // that isn't in a C-style comment or a string.
474 size_type slashPos(const string line, size_type start) {
475  if (start == npos)
476  return npos;
477 
478  size_type pos=line.find("//", start);
479  if (pos == npos)
480  return npos;
481 
482  // Skip over comments /* */ and strings both " " and ' '
483  size_type startComment=line.find("/*", start);
484  size_type startString=line.find_first_of("\'\"", start);
485 
486  if (min(startComment,startString) < pos)
487  return slashPos(line,
488  startComment < startString ?
489  endOfComment(line, startComment+2) :
490  endOfString(line[startString], line, startString+1));
491  else
492  // Nothing to skip over - the // actually starts a comment.
493  return pos;
494 }

496 string endCommentOrString(const string line) {
497  size_type pos=slashPos(line, 0);
498  if (pos == npos)
499  return endString(line, 0);
500  else {
501  string sub=line;
502 
503  // Replace the first // by /*
504  sub[pos+1]='*';
505 
506  // Replace any */ in the comment by *!
507  while ((pos = line.find("*/", pos+2)) != npos)
508  sub[pos+1]='!';
509 
510  // Tack on a */ at the end.
511  sub.append(" */ ");
512  return sub;
513  }
514 }

743  void runLine(coenv &e, istack &s, string line) {
744  try {
745  if(getSetting<bool>("multiline")) {
746  block *code=parseExtendableLine(line);
747 
748  icode i(code);
749  i.run(e,s,TRANS_INTERACTIVE);
750  } else if(settings::xasy) {
751  block *code=parseXasyLine(line);
752 
753  icode i(code);
754  i.run(e,s,TRANS_INTERACTIVE);
755  } else {
756  // Add a semi-colon to the end of the line if one is not there. Do this
757  // to the history as well, so the transcript can be run as regular asy
758  // code. This also makes the history work correctly if the multiline
759  // setting is changed within an interactive session.
760  // It is added to the history at the last possible moment to avoid
761  // tampering with other features, such as using a slash to extend a
762  // line.
