682 {

708 varEntry *venv::lookBySignature(symbol name, signature *sig) {
709  // Rest arguments are complicated and rare. Don't handle them here.
710  if (sig->hasRest())
711  return 0;
712 
713  // Likewise with the special operators.
714  if (name.special())
715  return 0;
716 
717  namevalue& nv = names[name];
718 
719  // Avoid ambiguities with default parameters.
720  if (nv.maxFormals != sig->getNumFormals())
721  return 0;
722 
723  // See if this exactly matches a function in the table.
724  varEntry *ve = core.lookupNonSpecial(name, sig);
725 
726  if (!ve)
727  return 0;
728 
729  // Keyword-only arguments may cause matching to fail.
730  if (ve->getSignature()->numKeywordOnly > 0)
731  return 0;
732

734  // to the result of the normal overloaded function resolution. We may
735  // safely return it.
736  return ve;
