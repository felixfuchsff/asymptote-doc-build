49 // This function attempts to produce a Bezier patch
50 // with the specified boundary path and normal directions.
51 // For instance, the patch should be normal to
52 // u0normals[0] at (0, 0.25),
53 // normal to u0normals[1] at (0, 0.5), and
54 // normal to u0normals[2] at (0, 0.75).
55 // The actual normal (as computed by the patch.normal() function)
56 // may be parallel to the specified normal, antiparallel, or
57 // even zero.
58 //
59 // A small amount of deviation is allowed in order to stabilize
60 // the algorithm (by keeping the mixed partials at the corners from
61 // growing too large).
62 //
63 // Note that the specified normals are projected to be orthogonal to
64 // the specified boundary path. However, the entries in the array
65 // remain intact.

233 // This function attempts to produce a Bezier triangle
234 // with the specified boundary path and normal directions at the
235 // edge midpoints. The bezier triangle should be normal to
236 // n1 at point(external, 0.5),
237 // normal to n2 at point(external, 1.5), and
238 // normal to n3 at point(external, 2.5).
239 // The actual normal (as computed by the patch.normal() function)
240 // may be parallel to the specified normal, antiparallel, or
241 // even zero.
242 //
243 // A small amount of deviation is allowed in order to stabilize
244 // the algorithm (by keeping the mixed partials at the corners from
245 // growing too large).
246 patch trianglewithnormals(path3 external, triple n1,
247  triple n2, triple n3) {
248  assert(cyclic(external));
249  assert(length(external) == 3);
250  // Use the formal symbols a3, a2b, abc, etc. to denote the control points,
251  // following the Wikipedia article on Bezier triangles.
252  triple a3 = point(external, 0), a2b = postcontrol(external, 0),
253  ab2 = precontrol(external, 1), b3 = point(external, 1),
254  b2c = postcontrol(external, 1), bc2 = precontrol(external, 2),
255  c3 = point(external, 2), ac2 = postcontrol(external, 2),
256  a2c = precontrol(external, 0);
257 
258  // Use orthogonal projection to ensure that the normal vectors are
259  // actually normal to the boundary path.
260  triple tangent = dir(external, 0.5);
261  n1 -= dot(n1,tangent)*tangent;
262  n1 = unit(n1);
263 
264  tangent = dir(external, 1.5);
265  n2 -= dot(n2,tangent)*tangent;
266  n2 = unit(n2);
267 
268  tangent = dir(external, 2.5);
269  n3 -= dot(n3,tangent)*tangent;
270  n3 = unit(n3);
271 
272  real wild = 2 * wildnessweight;
273  real[][] matrix = { {n1.x, n1.y, n1.z},
274  {n2.x, n2.y, n2.z},
275  {n3.x, n3.y, n3.z},
276  { wild, 0, 0},
277  { 0, wild, 0},
278  { 0, 0, wild} };
279  real[] rightvector =
280  { dot(n1, (a3 + 3a2b + 3ab2 + b3 - 2a2c - 2b2c)) / 4,
281  dot(n2, (b3 + 3b2c + 3bc2 + c3 - 2ab2 - 2ac2)) / 4,
282  dot(n3, (c3 + 3ac2 + 3a2c + a3 - 2bc2 - 2a2b)) / 4 };
283 
284  // The inner control point that minimizes the sum of squares of
285  // the mixed partials on the corners.
286  triple tameinnercontrol =
287  ((a2b + a2c - a3) + (ab2 + b2c - b3) + (ac2 + bc2 - c3)) / 3;
288  rightvector.append(wild * new real[]
289  {tameinnercontrol.x, tameinnercontrol.y, tameinnercontrol.z});
290  real[] solution = leastsquares(matrix, rightvector, warn=false);
291  if (solution.length == 0) { // if the matrix was singular
292  write("Warning: unable to solve matrix for specifying edge normals "
293  + "on bezier triangle. Using coons triangle.");
294  return patch(external);
295  }
296  triple innercontrol = (solution[0], solution[1], solution[2]);
297  return patch(external, innercontrol);
298 }

300 // A wrapper for the previous functions when the normal direction
301 // is given as a function of direction. The wrapper can also
302 // accommodate cyclic boundary paths of between one and four
303 // segments, although the results are best by far when there
304 // are three or four segments.
305 patch patchwithnormals(path3 external, triple normalat(triple)) {
306  assert(cyclic(external));
307  assert(1 <= length(external) && length(external) <= 4);
308  if (length(external) == 3) {
309  triple n1 = normalat(point(external, 0.5));
310  triple n2 = normalat(point(external, 1.5));
311  triple n3 = normalat(point(external, 2.5));
312  return trianglewithnormals(external, n1, n2, n3);
313  }
314  while (length(external) < 4) external = external -- cycle;
315  triple[] u0normals = new triple[3];
316  triple[] u1normals = new triple[3];
317  triple[] v0normals = new triple[3];
318  triple[] v1normals = new triple[3];
319  for (int i = 1; i <= 3; ++i) {
320  v0normals[i-1] = unit(normalat(point(external, i/4)));
321  u1normals[i-1] = unit(normalat(point(external, 1 + i/4)));
322  v1normals[i-1] = unit(normalat(point(external, 3 - i/4)));
323  u0normals[i-1] = unit(normalat(point(external, 4 - i/4)));
324  }
325  return patchwithnormals(external, u0normals, u1normals, v0normals, v1normals);
326 }

472 // Finally, the function that strings together a list of edges
473 // into a cycle. It makes assumptions that hold true if the
474 // list of edges did in fact come from a plane intersection
475 // containing no vertices of the cube. For instance, such a
476 // plane can contain at most two noncollinear points of any
477 // one face; consequently, no face can border more than two of
478 // the selected edges.
479 //
480 // If the underlying assumptions prove to be false, the function
481 // returns null.
482 int[] makecircle(edge[] edges) {
483  if (edges.length == 0) return new int[0];
484  int_to_intset graph;
485  for (edge e : edges) {
486  graph.add(e.start, e.end);
487  graph.add(e.end, e.start);
488  }
489  int currentvertex = edges[0].start;
490  int startvertex = currentvertex;
491  int lastvertex = NULL_VERTEX;
492  int[] toreturn = new int[0];
493  do {
494  toreturn.push(currentvertex);
495  int[] adjacentvertices = graph.get(currentvertex);
496  if (adjacentvertices.length != 2) return null;
497  for (int v : adjacentvertices) {
498  if (v != lastvertex) {
499  lastvertex = currentvertex;
500  currentvertex = v;
501  break;
502  }
503  }
504  } while (currentvertex != startvertex);
505  if (toreturn.length != graph.numkeys()) return null;
506  toreturn.cyclic = true;
507  return toreturn;
508 }

606 // This function applies a heuristic to choose two "opposite"
607 // vertices (separated by three segments) of edgecycle, which
608 // is required to be a cyclic path consisting of 5 or 6 segments.
609 // The two chosen vertices are pushed to savevertices.
610 //
611 // The function returns a path between the two chosen vertices. The
612 // path tangent at each endpoint is guaranteed to lie within the
613 // quarter-plane spanned by positive linear combinations of the
614 // tangents of the two outgoing paths at that endpoint.
615 path3 bisector(path3 edgecycle, int[] savevertices) {
616  real mincoeff = 0.05;
617  assert(cyclic(edgecycle));
618  int n = length(edgecycle);
619  assert(n >= 5 && n <= 6);
620  triple[] forwarddirections = sequence(new triple(int i) {
621  return dir(edgecycle, i, sign=1);
622  }, n);
623  forwarddirections.cyclic = true;
624  triple[] backwarddirections = sequence(new triple(int i) {
625  return -dir(edgecycle, i, sign=-1);
626  }, n);
627  backwarddirections.cyclic = true;
628  real[] angles = sequence(new real(int i) {
629  return angledegrees(forwarddirections[i], backwarddirections[i]);
630  }, n);
631  angles.cyclic = true;
632  int lastindex = (n == 5 ? 4 : 2);
633  real maxgoodness = 0;
634  int chosenindex = -1;
635  triple directionout, directionin;
636  for (int i = 0; i <= lastindex; ++i) {
637  int opposite = i + 3;
638  triple vec = unit(point(edgecycle, opposite) - point(edgecycle, i));
639  real[] coeffsbegin = projecttospan_findcoeffs(vec, forwarddirections[i],
640  backwarddirections[i]);
641  if (coeffsbegin.length == 0) continue;
642  coeffsbegin[0] = max(coeffsbegin[0], mincoeff);
643  coeffsbegin[1] = max(coeffsbegin[1], mincoeff);
644 
645  real[] coeffsend = projecttospan_findcoeffs(-vec, forwarddirections[opposite],
646  backwarddirections[opposite]);
647  if (coeffsend.length == 0) continue;
648  coeffsend[0] = max(coeffsend[0], mincoeff);
649  coeffsend[1] = max(coeffsend[1], mincoeff);
650 
651  real goodness = angles[i] * angles[opposite] * coeffsbegin[0] * coeffsend[0]
652  * coeffsbegin[1] * coeffsend[1];
653  if (goodness > maxgoodness) {
654  maxgoodness = goodness;
655  directionout = coeffsbegin[0] * forwarddirections[i] +
656  coeffsbegin[1] * backwarddirections[i];
657  directionin = -(coeffsend[0] * forwarddirections[opposite] +
658  coeffsend[1] * backwarddirections[opposite]);
659  chosenindex = i;
660  }
661  }
662  if (chosenindex == -1) {
663  savevertices.push(0);
664  savevertices.push(3);
665  return pathbetween(edgecycle, 0, 3);
666  } else {
667  savevertices.push(chosenindex);
668  savevertices.push(chosenindex+3);
669  return point(edgecycle, chosenindex) {directionout} ..
670  {directionin} point(edgecycle, chosenindex + 3);
671  }
672 }

718 // Quadrilateralization:
719 // Produce a surface (array of *nondegenerate* Bezier patches) with a
720 // specified three-segment boundary. The surface should approximate the
721 // zero locus of the specified f with its specified gradient.
722 //
723 // If it is not possible to produce the desired result without leaving the
724 // specified rectangular region, returns a length-zero array.
725 //
726 // Dividing a triangle into smaller quadrilaterals this way is opposite
727 // the usual trend in mathematics. However, *before the introduction of bezier
728 // triangles,* the pathwithnormals algorithm
729 // did a poor job of choosing a good surface when the boundary path did
730 // not consist of four positive-length segments.
731 patch[] triangletoquads(path3 external, real f(triple), triple grad(triple),
732  triple a, triple b) {
733  static real epsilon = 1e-3;
734  assert(length(external) == 3);
735  assert(cyclic(external));
736 
737  triple c0 = point(external, 0);
738  triple c1 = point(external, 1);
739  triple c2 = point(external, 2);
740 
741  triple center = (c0 + c1 + c2) / 3;
742  triple n = unit(cross(c1-c0, c2-c0));
743 
744  real g(real t) { return f(center + t*n); }
745 
746  real tmin = -realMax, tmax = realMax;
747  void absorb(real t) {
748  if (t < 0) tmin = max(t,tmin);
749  else tmax = min(t,tmax);
750  }
751  if (n.x != 0) {
752  absorb((a.x - center.x) / n.x);
753  absorb((b.x - center.x) / n.x);
754  }
755  if (n.y != 0) {
756  absorb((a.y - center.y) / n.y);
757  absorb((b.y - center.y) / n.y);
758  }
759  if (n.z != 0) {
760  absorb((a.z - center.z) / n.z);
761  absorb((b.z - center.z) / n.z);
762  }
763 
764  real fa = g(tmin);
765  real fb = g(tmax);
766  if ((fa > 0 && fb > 0) || (fa < 0 && fb < 0)) {
767  return new patch[0];
768  } else {
769  real t = findroot(g, tmin, tmax, fa=fa, fb=fb);
770  center += t * n;
771  }
772 
773  n = unit(grad(center));
774 
775  triple m0 = point(external, 0.5);
776  positionedvector m0 = positionedvector(m0, unit(grad(m0)));
777  triple m1 = point(external, 1.5);
778  positionedvector m1 = positionedvector(m1, unit(grad(m1)));
779  triple m2 = point(external, 2.5);
780  positionedvector m2 = positionedvector(m2, unit(grad(m2)));
781  positionedvector c = positionedvector(center, unit(grad(center)));
782 
783  path3 pathto_m0 = pathbetween(c, m0);
784  path3 pathto_m1 = pathbetween(c, m1);
785  path3 pathto_m2 = pathbetween(c, m2);

788  & reverse(pathto_m0)
789  & pathto_m2
790  & subpath(external, -0.5, 0)
791  & cycle;

793  & reverse(pathto_m1)
794  & pathto_m0
795  & subpath(external, 0.5, 1)
796  & cycle;
797  path3 quad2 = subpath(external, 2, 2.5)
798  & reverse(pathto_m2)
799  & pathto_m1
800  & subpath(external, 1.5, 2)
801  & cycle;
802 
803  return new patch[] {patchwithnormals(quad0, grad),
804  patchwithnormals(quad1, grad),
805  patchwithnormals(quad2, grad)};

808 // Attempts to fill the path external (which should by a cyclic path consisting of
809 // three segments) with bezier triangle(s). Returns an empty array if it fails.
810 //
811 // In more detail: A single bezier triangle is computed using trianglewithnormals. The normals of
812 // the resulting triangle at the midpoint of each edge are computed. If any of these normals
813 // is in the negative f direction, the external triangle is subdivided into four external triangles
814 // and the same procedure is applied to each. If one or more of them has an incorrectly oriented
815 // edge normal, the function gives up and returns an empty array.
816 //
817 // Thus, the returned array consists of 0, 1, or 4 bezier triangles; no other array lengths
818 // are possible.
819 //
820 // This function assumes that the path orientation is consistent with f (and its gradient)
821 // -- i.e., that
822 // at a corner, (tangent in) x (tangent out) is in the positive f direction.
823 patch[] maketriangle(path3 external, real f(triple),
824  triple grad(triple), bool allowsubdivide = true) {

826  assert(length(external) == 3);
827  triple m1 = point(external, 0.5);
828  triple n1 = unit(grad(m1));
829  triple m2 = point(external, 1.5);
830  triple n2 = unit(grad(m2));
831  triple m3 = point(external, 2.5);
832  triple n3 = unit(grad(m3));
833  patch beziertriangle = trianglewithnormals(external, n1, n2, n3);
834  if (dot(n1, beziertriangle.normal(0.5, 0)) >= 0 &&
835  dot(n2, beziertriangle.normal(0.5, 0.5)) >= 0 &&
836  dot(n3, beziertriangle.normal(0, 0.5)) >= 0)
837  return new patch[] {beziertriangle};
838 
839  if (!allowsubdivide) return new patch[0];
840 
841  positionedvector m1 = positionedvector(m1, n1);
842  positionedvector m2 = positionedvector(m2, n2);
843  positionedvector m3 = positionedvector(m3, n3);
844  path3 p12 = pathbetween(m1, m2);
845  path3 p23 = pathbetween(m2, m3);
846  path3 p31 = pathbetween(m3, m1);
847  patch[] triangles = maketriangle(p12 & p23 & p31 & cycle, f, grad=grad,
848  allowsubdivide=false);
849  if (triangles.length < 1) return new patch[0];
850 
851  triangles.append(maketriangle(subpath(external, -0.5, 0.5) & reverse(p31) & cycle,
852  f, grad=grad, allowsubdivide=false));
853  if (triangles.length < 2) return new patch[0];
854 
855  triangles.append(maketriangle(subpath(external, 0.5, 1.5) & reverse(p12) & cycle,
856  f, grad=grad, allowsubdivide=false));
857  if (triangles.length < 3) return new patch[0];

860  f, grad=grad, allowsubdivide=false));

915  static real tolerance = 2.5 * rootfinder_settings.roottolerance;

918  // unify them into one.

947  dir(edgecycle,i+1,sign=-1)) > 80) {
