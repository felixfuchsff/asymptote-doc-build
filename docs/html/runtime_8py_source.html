148 def parse_types(
149  data: str, input_filename: str, input_start_line: int
150 ) -> tuple[dict[str, str], int]:
151  """Return (new_entries, lines_consumed)."""
152  entries: dict[str, str] = {}
153  lines = data.split("\n")
154  for line_no, line in enumerate(lines, start=input_start_line):
155  # Remove // comments in each line
156  trimmed = re.sub(r"//.*", "", line)
157 
158  # Skip empty lines
159  if re.fullmatch(r"\s*", trimmed):
160  continue
161  # Regex for matching something like "item => ITEMVAL"
162  # (\w*(?:\s*\*)?) matches a type or pointer (e.g. "int", "item*", etc.)
163  # \s*=>\s* captures the '=>' then
164  # (.*) grabs the rest of the line as the code
165  match = re.search(
166  r"(\w*(?:\s*\*)?)" # Type or pointer type
167  r"\s*=>\s*" # =>
168  r"(.*)", # Everything else
169  trimmed,
170  )
171  if not match:
172  Er.report_error(input_filename, line_no, "bad type declaration")
173  continue
174  t, code = match.groups()
175  entries[clean_type(t)] = code
176  return entries, data.count("\n")
