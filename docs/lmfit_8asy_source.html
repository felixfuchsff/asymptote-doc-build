25 /*
26  Fitting $n$ data points $(x_1, y_1 \pm \Delta y_1), \dots, (x_n, y_n \pm \Delta y_n)$
27  to a function $f$ that depends on $m$ parameters $a_1, \dots, a_m$ means minimizing
28  the least-squares sum
29  %
30  \begin{equation*}
31  \sum_{i = 1}^n \left( \frac{y_i - f(a_1, \dots, a_m; x_i)}{\Delta y_i} \right)^2
32  \end{equation*}
33  %
34  with respect to the parameters $a_1, \dots, a_m$.
35 */
36 
37 /*
38  This module provides an implementation of the Levenberg--Marquardt
39  (LM) algorithm, converted from the C lmfit routine by Joachim Wuttke
40  (see http://www.messen-und-deuten.de/lmfit/).
41 
42  Implementation strategy: Fortunately, Asymptote's syntax is very
43  similar to C, and the original code cleanly separates the
44  customizable parts (user-provided data, output routines, etc.) from
45  the dirty number crunching. Thus, mst of the code was just copied
46  and slightly modified from the original source files. I have
47  amended the lm_data_type structure and the callback routines with a
48  weight array that can be used to provide experimental errors. I
49  have also created two simple wrapper functions.
50 */

153 // Asymptote doesn't support pointers to arbitrary array elements, so
154 // we provide an offset parameter.
155 private real lm_enorm(int n, real[] x, int offset=0) {
156  real s1 = 0;
157  real s2 = 0;
158  real s3 = 0;
159  real x1max = 0;
160  real x3max = 0;
161  real agiant = LM_SQRT_GIANT / n;
162  real xabs, temp;
163 
164  for (int i = 0; i < n; ++i) {
165  xabs = fabs(x[offset + i]);
166  if (xabs > LM_SQRT_DWARF && xabs < agiant) {
167  s2 += SQR(xabs);
168  continue;
169  }
170 
171  if (xabs > LM_SQRT_DWARF) {
172  if (xabs > x1max) {
173  temp = x1max / xabs;
174  s1 = 1 + s1 * SQR(temp);
175  x1max = xabs;
176  } else {
177  temp = xabs / x1max;
178  s1 += SQR(temp);
179  }
180  continue;
181  }
182  if (xabs > x3max) {
183  temp = x3max / xabs;
184  s3 = 1 + s3 * SQR(temp);
185  x3max = xabs;
186  } else {
187  if (xabs != 0.0) {
188  temp = xabs / x3max;
189  s3 += SQR(temp);
190  }
191  }
192  }
193 
194  if (s1 != 0)
195  return x1max * sqrt(s1 + (s2 / x1max) / x1max);
196  if (s2 != 0) {
197  if (s2 >= x3max)
198  return sqrt(s2 * (1 + (x3max / s2) * (x3max * s3)));
199  else
200  return sqrt(x3max * ((s2 / x3max) + (x3max * s3)));
201  }
202 
203  return x3max * sqrt(s3);
204 }

277 // copied from the C code
278 private void lm_qrfac(int m, int n, real[] a, bool pivot, int[] ipvt, real[] rdiag, real[] acnorm, real[] wa) {
279  int i, j, k, kmax, minmn;
280  real ajnorm, sum, temp;
281  static real p05 = 0.05;
282 
283  for (j = 0; j < n; ++j) {
284  acnorm[j] = lm_enorm(m, a, j * m);
285  rdiag[j] = acnorm[j];
286  wa[j] = rdiag[j];
287  if (pivot)
288  ipvt[j] = j;
289  }
290 
291  minmn = min(m, n);
292  for (j = 0; j < minmn; ++j) {
293  while (pivot) {
294  kmax = j;
295  for (k = j + 1; k < n; ++k)
296  if (rdiag[k] > rdiag[kmax])
297  kmax = k;
298  if (kmax == j)
299  break;
300 
301  for (i = 0; i < m; ++i) {
302  temp = a[j * m + i];
303  a[j * m + i] = a[kmax * m + i];
304  a[kmax * m + i] = temp;
305  }
306  rdiag[kmax] = rdiag[j];
307  wa[kmax] = wa[j];
308  k = ipvt[j];
309  ipvt[j] = ipvt[kmax];
310  ipvt[kmax] = k;
311 
312  break;
313  }
314 
315  ajnorm = lm_enorm(m - j, a, j * m + j);
316  if (ajnorm == 0.0) {
317  rdiag[j] = 0;
318  continue;
319  }
320 
321  if (a[j * m + j] < 0.0)
322  ajnorm = -ajnorm;
323  for (i = j; i < m; ++i)
324  a[j * m + i] /= ajnorm;
325  a[j * m + j] += 1;
326 
327  for (k = j + 1; k < n; ++k) {
328  sum = 0;
329 
330  for (i = j; i < m; ++i)
331  sum += a[j * m + i] * a[k * m + i];
332 
333  temp = sum / a[j + m * j];
334 
335  for (i = j; i < m; ++i)
336  a[k * m + i] -= temp * a[j * m + i];
337 
338  if (pivot && rdiag[k] != 0.0) {
339  temp = a[m * k + j] / rdiag[k];
340  temp = max(0.0, 1 - SQR(temp));
341  rdiag[k] *= sqrt(temp);
342  temp = rdiag[k] / wa[k];
343  if (p05 * SQR(temp) <= LM_MACHEP) {
344  rdiag[k] = lm_enorm(m - j - 1, a, m * k + j + 1);
345  wa[k] = rdiag[k];
346  }
347  }
348  }
349 
350  rdiag[j] = -ajnorm;
351  }
352 }

356 private void lm_qrsolv(int n, real[] r, int ldr, int[] ipvt, real[] diag, real[] qtb, real[] x, real[] sdiag, real[] wa) {
357  static real p25 = 0.25;
358  static real p5 = 0.5;
359 
360  int i, kk, j, k, nsing;
361  real qtbpj, sum, temp;
362  real _sin, _cos, _tan, _cot;
363 
364  for (j = 0; j < n; ++j) {
365  for (i = j; i < n; ++i)
366  r[j * ldr + i] = r[i * ldr + j];
367  x[j] = r[j * ldr + j];
368  wa[j] = qtb[j];
369  }
370 
371  for (j = 0; j < n; ++j) {
372  while (diag[ipvt[j]] != 0.0) {
373  for (k = j; k < n; ++k)
374  sdiag[k] = 0.0;
375  sdiag[j] = diag[ipvt[j]];
376 
377  qtbpj = 0.;
378  for (k = j; k < n; ++k) {
379  if (sdiag[k] == 0.)
380  continue;
381  kk = k + ldr * k;
382  if (fabs(r[kk]) < fabs(sdiag[k])) {
383  _cot = r[kk] / sdiag[k];
384  _sin = p5 / sqrt(p25 + p25 * _cot * _cot);
385  _cos = _sin * _cot;
386  } else {
387  _tan = sdiag[k] / r[kk];
388  _cos = p5 / sqrt(p25 + p25 * _tan * _tan);
389  _sin = _cos * _tan;
390  }
391 
392  r[kk] = _cos * r[kk] + _sin * sdiag[k];
393  temp = _cos * wa[k] + _sin * qtbpj;
394  qtbpj = -_sin * wa[k] + _cos * qtbpj;
395  wa[k] = temp;
396 
397  for (i = k + 1; i < n; ++i) {
398  temp = _cos * r[k * ldr + i] + _sin * sdiag[i];
399  sdiag[i] = -_sin * r[k * ldr + i] + _cos * sdiag[i];
400  r[k * ldr + i] = temp;
401  }
402  }
403  break;
404  }
405 
406  sdiag[j] = r[j * ldr + j];
407  r[j * ldr + j] = x[j];
408  }
409 
410  nsing = n;
411  for (j = 0; j < n; ++j) {
412  if (sdiag[j] == 0.0 && nsing == n)
413  nsing = j;
414  if (nsing < n)
415  wa[j] = 0;
416  }
417 
418  for (j = nsing - 1; j >= 0; --j) {
419  sum = 0;
420  for (i = j + 1; i < nsing; ++i)
421  sum += r[j * ldr + i] * wa[i];
422  wa[j] = (wa[j] - sum) / sdiag[j];
423  }
424 
425  for (j = 0; j < n; ++j)
426  x[ipvt[j]] = wa[j];
427 }

430 // copied from the C code
431 private void lm_lmpar(int n, real[] r, int ldr, int[] ipvt, real[] diag, real[] qtb, real delta, lm_real_type par, real[] x, real[] sdiag, real[] wa1, real[] wa2) {
432  static real p1 = 0.1;
433  static real p001 = 0.001;
434 
435  int nsing = n;
436  real parl = 0.0;
437 
438  int i, iter, j;
439  real dxnorm, fp, fp_old, gnorm, parc, paru;
440  real sum, temp;
441 
442  for (j = 0; j < n; ++j) {
443  wa1[j] = qtb[j];
444  if (r[j * ldr + j] == 0 && nsing == n)
445  nsing = j;
446  if (nsing < n)
447  wa1[j] = 0;
448  }
449  for (j = nsing - 1; j >= 0; --j) {
450  wa1[j] = wa1[j] / r[j + ldr * j];
451  temp = wa1[j];
452  for (i = 0; i < j; ++i)
453  wa1[i] -= r[j * ldr + i] * temp;
454  }
455 
456  for (j = 0; j < n; ++j)
457  x[ipvt[j]] = wa1[j];
458 
459  iter = 0;
460  for (j = 0; j < n; ++j)
461  wa2[j] = diag[j] * x[j];
462  dxnorm = lm_enorm(n, wa2);
463  fp = dxnorm - delta;
464  if (fp <= p1 * delta) {
465  par.val = 0;
466  return;
467  }
468 
469  if (nsing >= n) {
470  for (j = 0; j < n; ++j)
471  wa1[j] = diag[ipvt[j]] * wa2[ipvt[j]] / dxnorm;
472 
473  for (j = 0; j < n; ++j) {
474  sum = 0.0;
475  for (i = 0; i < j; ++i)
476  sum += r[j * ldr + i] * wa1[i];
477  wa1[j] = (wa1[j] - sum) / r[j + ldr * j];
478  }
479  temp = lm_enorm(n, wa1);
480  parl = fp / delta / temp / temp;
481  }
482 
483  for (j = 0; j < n; ++j) {
484  sum = 0;
485  for (i = 0; i <= j; ++i)
486  sum += r[j * ldr + i] * qtb[i];
487  wa1[j] = sum / diag[ipvt[j]];
488  }
489  gnorm = lm_enorm(n, wa1);
490  paru = gnorm / delta;
491  if (paru == 0.0)
492  paru = LM_DWARF / min(delta, p1);
493 
494  par.val = max(par.val, parl);
495  par.val = min(par.val, paru);
496  if (par.val == 0.0)
497  par.val = gnorm / dxnorm;
498 
499  for (;; ++iter) {
500  if (par.val == 0.0)
501  par.val = max(LM_DWARF, p001 * paru);
502  temp = sqrt(par.val);
503  for (j = 0; j < n; ++j)
504  wa1[j] = temp * diag[j];
505  lm_qrsolv(n, r, ldr, ipvt, wa1, qtb, x, sdiag, wa2);
506  for (j = 0; j < n; ++j)
507  wa2[j] = diag[j] * x[j];
508  dxnorm = lm_enorm(n, wa2);
509  fp_old = fp;
510  fp = dxnorm - delta;
511 
512  if (fabs(fp) <= p1 * delta || (parl == 0.0 && fp <= fp_old && fp_old < 0.0) || iter == 10)
513  break;
514 
515  for (j = 0; j < n; ++j)
516  wa1[j] = diag[ipvt[j]] * wa2[ipvt[j]] / dxnorm;
517 
518  for (j = 0; j < n; ++j) {
519  wa1[j] = wa1[j] / sdiag[j];
520  for (i = j + 1; i < n; ++i)
521  wa1[i] -= r[j * ldr + i] * wa1[j];
522  }
523  temp = lm_enorm(n, wa1);
524  parc = fp / delta / temp / temp;
525 
526  if (fp > 0)
527  parl = max(parl, par.val);
528  else if (fp < 0)
529  paru = min(paru, par.val);
530 
531  par.val = max(parl, par.val + parc);
532  }
533 }

537 void lm_lmdif(int m, int n, real[] x, real[] fvec, real ftol, real xtol, real gtol, int maxfev, real epsfcn, real[] diag, int mode, real factor, lm_int_type info, lm_int_type nfev, real[] fjac, int[] ipvt, real[] qtf, real[] wa1, real[] wa2, real[] wa3, real[] wa4, lm_evaluate_ftype evaluate, lm_print_ftype printout, lm_data_type data) {
538  static real p1 = 0.1;
539  static real p5 = 0.5;
540  static real p25 = 0.25;
541  static real p75 = 0.75;
542  static real p0001 = 1.0e-4;
543 
544  nfev.val = 0;
545  int iter = 1;
546  lm_real_type par = lm_real_type(0);
547  real delta = 0;
548  real xnorm = 0;
549  real temp = max(epsfcn, LM_MACHEP);
550  real eps = sqrt(temp);
551  int i, j;
552  real actred, dirder, fnorm, fnorm1, gnorm, pnorm, prered, ratio, step, sum, temp1, temp2, temp3;
553 
554  if ((n <= 0) || (m < n) || (ftol < 0.0) || (xtol < 0.0) || (gtol < 0.0) || (maxfev <= 0) || (factor <= 0)) {
555  info.val = 0;
556  return;
557  }
558  if (mode == 2) {
559  for (j = 0; j < n; ++j) {
560  if (diag[j] <= 0.0) {
561  info.val = 0;
562  return;
563  }
564  }
565  }
566 
567  info.val = 0;
568  evaluate(x, m, fvec, data, info);
569  if(printout != null) printout(n, x, m, fvec, data, 0, 0, ++nfev.val);
570  if (info.val < 0)
571  return;
572  fnorm = lm_enorm(m, fvec);
573 
574  do {
575  for (j = 0; j < n; ++j) {
576  temp = x[j];
577  step = eps * fabs(temp);
578  if (step == 0.0)
579  step = eps;
580  x[j] = temp + step;
581  info.val = 0;
582  evaluate(x, m, wa4, data, info);
583  if(printout != null) printout(n, x, m, wa4, data, 1, iter, ++nfev.val);
584  if (info.val < 0)
585  return;
586  for (i = 0; i < m; ++i)
587  fjac[j * m + i] = (wa4[i] - fvec[i]) / (x[j] - temp);
588  x[j] = temp;
589  }
590 
591  lm_qrfac(m, n, fjac, true, ipvt, wa1, wa2, wa3);
592 
593  if (iter == 1) {
594  if (mode != 2) {
595  for (j = 0; j < n; ++j) {
596  diag[j] = wa2[j];
597  if (wa2[j] == 0.0)
598  diag[j] = 1.0;
599  }
600  }
601  for (j = 0; j < n; ++j)
602  wa3[j] = diag[j] * x[j];
603  xnorm = lm_enorm(n, wa3);
604  delta = factor * xnorm;
605  if (delta == 0.0)
606  delta = factor;
607  }
608 
609  for (i = 0; i < m; ++i)
610  wa4[i] = fvec[i];
611 
612  for (j = 0; j < n; ++j) {
613  temp3 = fjac[j * m + j];
614  if (temp3 != 0.0) {
615  sum = 0;
616  for (i = j; i < m; ++i)
617  sum += fjac[j * m + i] * wa4[i];
618  temp = -sum / temp3;
619  for (i = j; i < m; ++i)
620  wa4[i] += fjac[j * m + i] * temp;
621  }
622  fjac[j * m + j] = wa1[j];
623  qtf[j] = wa4[j];
624  }
625 
626  gnorm = 0;
627  if (fnorm != 0) {
628  for (j = 0; j < n; ++j) {
629  if (wa2[ipvt[j]] == 0) continue;
630  sum = 0.0;
631  for (i = 0; i <= j; ++i)
632  sum += fjac[j * m + i] * qtf[i] / fnorm;
633  gnorm = max(gnorm, fabs(sum / wa2[ipvt[j]]));
634  }
635  }
636 
637  if (gnorm <= gtol) {
638  info.val = 4;
639  return;
640  }
641 
642  if (mode != 2) {
643  for (j = 0; j < n; ++j)
644  diag[j] = max(diag[j], wa2[j]);
645  }
646 
647  do {
648  lm_lmpar(n, fjac, m, ipvt, diag, qtf, delta, par, wa1, wa2, wa3, wa4);
649 
650  for (j = 0; j < n; ++j) {
651  wa1[j] = -wa1[j];
652  wa2[j] = x[j] + wa1[j];
653  wa3[j] = diag[j] * wa1[j];
654  }
655  pnorm = lm_enorm(n, wa3);
656 
657  if (nfev.val <= 1 + n)
658  delta = min(delta, pnorm);
659 
660  info.val = 0;
661  evaluate(wa2, m, wa4, data, info);
662  if(printout != null) printout(n, x, m, wa4, data, 2, iter, ++nfev.val);
663  if (info.val < 0)
664  return;
665 
666  fnorm1 = lm_enorm(m, wa4);
667 
668  if (p1 * fnorm1 < fnorm)
669  actred = 1 - SQR(fnorm1 / fnorm);
670  else
671  actred = -1;
672 
673  for (j = 0; j < n; ++j) {
674  wa3[j] = 0;
675  for (i = 0; i <= j; ++i)
676  wa3[i] += fjac[j * m + i] * wa1[ipvt[j]];
677  }
678  temp1 = lm_enorm(n, wa3) / fnorm;
679  temp2 = sqrt(par.val) * pnorm / fnorm;
680  prered = SQR(temp1) + 2 * SQR(temp2);
681  dirder = -(SQR(temp1) + SQR(temp2));
682 
683  ratio = prered != 0 ? actred / prered : 0;
684 
685  if (ratio <= p25) {
686  if (actred >= 0.0)
687  temp = p5;
688  else
689  temp = p5 * dirder / (dirder + p5 * actred);
690  if (p1 * fnorm1 >= fnorm || temp < p1)
691  temp = p1;
692  delta = temp * min(delta, pnorm / p1);
693  par.val /= temp;
694  } else if (par.val == 0.0 || ratio >= p75) {
695  delta = pnorm / p5;
696  par.val *= p5;
697  }
698 
699  if (ratio >= p0001) {
700  for (j = 0; j < n; ++j) {
701  x[j] = wa2[j];
702  wa2[j] = diag[j] * x[j];
703  }
704  for (i = 0; i < m; ++i)
705  fvec[i] = wa4[i];
706  xnorm = lm_enorm(n, wa2);
707  fnorm = fnorm1;
708  ++iter;
709  }
710 
711  info.val = 0;
712  if (fabs(actred) <= ftol && prered <= ftol && p5 * ratio <= 1)
713  info.val = 1;
714  if (delta <= xtol * xnorm)
715  info.val += 2;
716  if (info.val != 0)
717  return;
718 
719  if (nfev.val >= maxfev)
720  info.val = 5;
721  if (fabs(actred) <= LM_MACHEP && prered <= LM_MACHEP && p5 * ratio <= 1)
722  info.val = 6;
723  if (delta <= LM_MACHEP * xnorm)
724  info.val = 7;
725  if (gnorm <= LM_MACHEP)
726  info.val = 8;
727  if (info.val != 0)
728  return;
729  } while (ratio < p0001);
730  } while (true);
731 }
