120 void drawLabel::bounds(bbox& b, iopipestream& tex, boxvector& labelbounds,
121  bboxlist&)
122 {
123  string texengine=getSetting<string>("tex");
124  if(texengine == "none") {b += position; return;}
125 
126  getbounds(tex,texengine);
127 
128  // alignment point
129  pair p=position+Align;
130  const double vertical=height+depth;
131  const double fuzz=pentype.size()*0.1+0.3;
132  pair A=p+T*pair(-fuzz,-fuzz);
133  pair B=p+T*pair(-fuzz,vertical+fuzz);
134  pair C=p+T*pair(width+fuzz,vertical+fuzz);
135  pair D=p+T*pair(width+fuzz,-fuzz);
136 
137  if(pentype.Overwrite() != ALLOW && label != "") {
138  size_t n=labelbounds.size();
139  box Box=box(A,B,C,D);
140  for(size_t i=0; i < n; i++) {
141  if(labelbounds[i].intersect(Box)) {
142  switch(pentype.Overwrite()) {
143  case SUPPRESS:
144  labelwarning("suppressed");
145  case SUPPRESSQUIET:
146  suppress=true;
147  return;
148  case MOVE:
149  labelwarning("moved");
150  default:
151  break;
152  }
153 
154  pair Align=(align == pair(0,0)) ? unit(pair(urand(),urand())) :
155  unit(align);
156  double s=0.1*pentype.size();
157  double dx=0, dy=0;
158  if(Align.getx() > 0.1) dx=labelbounds[i].xmax()-Box.xmin()+s;
159  if(Align.getx() < -0.1) dx=labelbounds[i].xmin()-Box.xmax()-s;
160  if(Align.gety() > 0.1) dy=labelbounds[i].ymax()-Box.ymin()+s;
161  if(Align.gety() < -0.1) dy=labelbounds[i].ymin()-Box.ymax()-s;
162  pair offset=pair(dx,dy);
163  position += offset;
164  A += offset;
165  B += offset;
166  C += offset;
167  D += offset;
168  Box=box(A,B,C,D);
169  i=0;
170  }
171  }
172  labelbounds.resize(n+1);
173  labelbounds[n]=Box;
174  }
175 
176  Box=bbox();
177  Box += A;
178  Box += B;
179  Box += C;
180  Box += D;
181

183 }
