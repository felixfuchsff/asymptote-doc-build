74 void psfile::dealias(unsigned char *a, size_t width, size_t height, size_t n,
75  bool convertrgb, ColorSpace colorspace)
76 {
77  // Dealias all but the last row and column of pixels.
78  size_t istop=width-1;
79  size_t jstop=height-1;
80  if(convertrgb) {
81  size_t nwidth=3*width;
82  for(size_t j=0; j < height; ++j) {
83  unsigned char *aj=a+nwidth*j;
84  for(size_t i=0; i < width; ++i) {
85  unsigned char *ai=aj+3*i;
86  if(i < istop && j < jstop)
87  writefromRGB(average(ai,3,nwidth),
88  average(ai+1,3,nwidth),
89  average(ai+2,3,nwidth),colorspace,n);
90  else
91  writefromRGB(ai[0],ai[1],ai[2],colorspace,n);
92  }
93  }
94  } else {
95  size_t nwidth=n*width;
96  for(size_t j=0; j < jstop; ++j) {
97  unsigned char *aj=a+nwidth*j;
98  for(size_t i=0; i < istop; ++i) {
99  unsigned char *ai=aj+n*i;
100  for(size_t k=0; k < n; ++k)
101  ai[k]=average(ai+k,n,nwidth);
102  }
103  }
104  }

451 void psfile::tensorshade(const pen& pentype, const array& pens,
452  const array& boundaries, const array& z)
453 {
454  checkLevel();
455  endclip(pentype);
456 
457  size_t size=pens.size();
458  if(size == 0) return;
459  size_t nz=z.size();
460 
461  array *p0=read<array *>(pens,0);
462  if(checkArray(p0) != 4)
463  reportError("4 pens required");
464  setfirstopacity(*p0);
465 
466  ColorSpace colorspace=maxcolorspace2(pens);
467  checkColorSpace(colorspace);
468 
469  *out << "<< /ShadingType 7" << newl
470  << "/ColorSpace /Device" << ColorDeviceSuffix[colorspace] << newl
471  << "/DataSource [" << newl;
472 
473  for(size_t i=0; i < size; i++) {
474  // Only edge flag 0 (new patch) is implemented since the 32% data
475  // compression (for RGB) afforded by other edge flags really isn't worth
476  // the trouble or confusion for the user.
477  write(0);
478  path g=read<path>(boundaries,i);
479  if(!(g.cyclic() && g.size() == 4))
480  reportError("specify cyclic path of length 4");
481  for(Int j=4; j > 0; --j) {
482  write(g.point(j));
483  write(g.precontrol(j));
484  write(g.postcontrol(j-1));
485  }
486  if(nz == 0) { // Coons patch
487  static double nineth=1.0/9.0;
488  for(Int j=0; j < 4; ++j) {
489  write(nineth*(-4.0*g.point(j)+6.0*(g.precontrol(j)+g.postcontrol(j))
490  -2.0*(g.point(j-1)+g.point(j+1))
491  +3.0*(g.precontrol(j-1)+g.postcontrol(j+1))
492  -g.point(j+2)));
493  }
494  } else {
495  array *zi=read<array *>(z,i);
496  if(checkArray(zi) != 4)
497  reportError("specify 4 internal control points for each path");

571 void psfile::image(const array& a, const array& P, bool antialias)
572 {
573  size_t asize=a.size();
574  size_t Psize=P.size();
575  if(asize == 0 || Psize == 0) return;
576 
577  array *a0=read<array *>(a,0);
578  size_t a0size=a0->size();
579  if(a0size == 0) return;
580 
581  setfirstopacity(P);
582 
583  ColorSpace colorspace=maxcolorspace(P);
584  checkColorSpace(colorspace);
585 
586  size_t ncomponents=ColorComponents[colorspace];
587 
588  imageheader(a0size,asize,colorspace);
589 
590  double min=read<double>(a0,0);
591  double max=min;
592  for(size_t i=0; i < asize; i++) {
593  array *ai=read<array *>(a,i);
594  size_t size=ai->size();
595  if(size != a0size)
596  reportError(rectangular);
597  for(size_t j=0; j < size; j++) {
598  double val=read<double>(ai,j);
599  if(val > max) max=val;
600  else if(val < min) min=val;
601  }
602  }
603 
604  double step=(max == min) ? 0.0 : (Psize-1)/(max-min);
605 
606  beginImage(ncomponents*a0size*asize);
607  for(size_t i=0; i < asize; i++) {
608  array *ai=read<array *>(a,i);
609  for(size_t j=0; j < a0size; j++) {
610  double val=read<double>(ai,j);
611  size_t index=(size_t) ((val-min)*step+0.5);
612  pen *p=read<pen *>(P,index < Psize ? index : Psize-1);
613  p->convert();
614  if(!p->promote(colorspace))
615  reportError(inconsistent);
616  write(p,ncomponents);
617  }
618  }
619  endImage(antialias,a0size,asize,ncomponents);
620 }
